could_be_temporal_religious_head_trigger = {
	trigger_if = {
		limit = { controls_religion = yes }

		OR = {
			religion_group = muslim

			trigger_if = {
				limit = { has_religion_features = yes } # TODO: Divine Ancestry
				has_religion_feature = religion_temporal_head
			}
			trigger_else = {
				religion = norse_pagan_reformed
			}
			
			religion = shinto
			religion = confucian # If they get a rel head down the line
			religion = taoist # If they get a rel head down the line
		}
	}
}

has_sympathy_for_true_religion_of_prev_trigger = {
	trigger_if = {
		limit = { PREV = { true_religion_group = christian } }
		trait = sympathy_christendom
	}
	trigger_else_if = {
		limit = { PREV = { true_religion_group = muslim } }
		trait = sympathy_islam
	}
	trigger_else_if = {
		limit = { PREV = { true_religion_group = pagan_group } }
		trait = sympathy_pagans
	}
	trigger_else_if = {
		limit = { PREV = { true_religion_group = zoroastrian_group } }
		trait = sympathy_zoroastrianism
	}
	trigger_else_if = {
		limit = { PREV = { true_religion_group = jewish_group } }
		trait = sympathy_judaism
	}
	trigger_else = {
		PREV = { true_religion_group = indian_group }
		trait = sympathy_indian
	}
}

has_sympathy_for_true_religion_of_root_trigger = {
	trigger_if = {
		limit = { ROOT = { true_religion_group = christian } }

has_safe_religion = {
	show_only_failed_conditions = yes

	trigger_if = {
		limit = { religion_group = jewish_group }
		has_dlc = "Sons of Abraham"
	}
	trigger_else_if = {
		limit = { religion_group = muslim }
		has_dlc = "The Sword of Islam"
	}
	trigger_else_if = {
		limit = { religion_openly_bon_or_reformed_trigger = yes }

		OR = {
			has_dlc = "Jade Dragon"
			has_dlc = "Holy Fury"
			has_dlc = "The Old Gods"
		}
	}
	trigger_else_if = {
		limit = { religion_group = pagan_group }

		OR = {
			has_dlc = "The Old Gods"
			has_dlc = "Holy Fury"
		}
	}
	trigger_else_if = {
		limit = { religion_group = zoroastrian_group }
		has_dlc = "The Old Gods"
	}
	trigger_else_if = {
		limit = { religion = taoist }

		OR = {
			has_dlc = "Rajas of India"
			has_dlc = "Jade Dragon"
		}
	}
	trigger_else_if = {
		limit = { religion_group = indian_group }
		has_dlc = "Rajas of India"
	}
}
