# inheritance_sanity_check_effect
# inheritance_sanity_cleanup_effect
# inheritance_sanity_setup_effect

inheritance_sanity_check_effect = { # Handles potential cases where traits/bloodlines don't respect (perceived) parents... and Jimmu's bloodline accidentally being universal due to vanilla stupidity...
	inheritance_sanity_cleanup_effect = yes
	inheritance_sanity_setup_effect = yes
}

inheritance_sanity_cleanup_effect = {
	hidden_tooltip = {
		remove_trait = sayyid
		remove_trait = mirza
		remove_trait = saoshyant_descendant
		remove_trait = amaterasu_descendant
		if = {
			limit = {
				any_owned_bloodline = {
					has_bloodline_flag = imperial_family
				}
				father_even_if_dead = {
					NOT = {
						any_owned_bloodline = {
							has_bloodline_flag = imperial_family
						}
					}
				}
			}
			any_owned_bloodline = {
				limit = {
					has_bloodline_flag = imperial_family
				}
				remove_bloodline_member = ROOT
			}
		}
		if = {
			limit = {
				any_owned_bloodline = {
					has_bloodline_flag = confucius
				}
				father_even_if_dead = {
					NOT = {
						any_owned_bloodline = {
							has_bloodline_flag = confucius
						}
					}
				}
			}
			any_owned_bloodline = {
				limit = {
					has_bloodline_flag = confucius
				}
				remove_bloodline_member = ROOT
			}
		}
		if = {
			limit = {
				any_owned_bloodline = {
					has_bloodline_flag = divine_ancestry_bloodline
				}
			}
			any_owned_bloodline = {
				limit = {
					has_bloodline_flag = divine_ancestry_bloodline
				}
				NOT = {
					any_bloodline_member_including_dead = {
						any_child = {
							character = ROOT
						}
					}
				}
				remove_bloodline_member = ROOT
			}
		}
	}
}

inheritance_sanity_setup_effect = {
	hidden_tooltip = {
		if = {
			limit = {
				father_even_if_dead = {
					trait = sayyid
				}
			}
			add_trait = sayyid
		}
		else_if = {
			limit = {
				mother_even_if_dead = {
					trait = sayyid
				}
			}
			add_trait = mirza
		}
		if = {
			limit = {
				OR = {
					father_even_if_dead = {
						OR = {
							trait = saoshyant
							trait = saoshyant_descendant
						}
					}
					mother_even_if_dead = {
						OR = {
							trait = saoshyant
							trait = saoshyant_descendant
						}
					}
				}
			}
			add_trait = saoshyant_descendant
		}
		if = {
			limit = {
				OR = {
					father_even_if_dead = {
						trait = amaterasu_descendant
					}
					mother_even_if_dead = {
						trait = amaterasu_descendant
					}
				}
			}
			add_trait = amaterasu_descendant
		}
		if = {
			limit = {
				father_even_if_dead = {
					any_owned_bloodline = {
						has_bloodline_flag = imperial_family
						NOT = {
							any_bloodline_member = {
								character = ROOT
							}
						}
					}
				}
			}
			father_even_if_dead = {
				any_owned_bloodline = {
					limit = {
						has_bloodline_flag = imperial_family
					}
					add_bloodline_member = ROOT
				}
			}
		}
		if = {
			limit = {
				father_even_if_dead = {
					any_owned_bloodline = {
						has_bloodline_flag = confucius
						NOT = {
							any_bloodline_member = {
								character = ROOT
							}
						}
					}
				}
			}
			father_even_if_dead = {
				any_owned_bloodline = {
					limit = {
						has_bloodline_flag = confucius
					}
					add_bloodline_member = ROOT
				}
			}
		}
		if = {
			limit = {
				father_even_if_dead = {
					any_owned_bloodline = {
						has_bloodline_flag = divine_ancestry_bloodline
						NOT = {
							any_bloodline_member = {
								character = ROOT
							}
						}
					}
				}
			}
			father_even_if_dead = {
				any_owned_bloodline = {
					limit = {
						has_bloodline_flag = divine_ancestry_bloodline
					}
					add_bloodline_member = ROOT
				}
			}
		}
		if = {
			limit = {
				mother_even_if_dead = {
					any_owned_bloodline = {
						has_bloodline_flag = divine_ancestry_bloodline
						NOT = {
							any_bloodline_member = {
								character = ROOT
							}
						}
					}
				}
			}
			mother_even_if_dead = {
				any_owned_bloodline = {
					limit = {
						has_bloodline_flag = divine_ancestry_bloodline
					}
					add_bloodline_member = ROOT
				}
			}
		}
	}
}
