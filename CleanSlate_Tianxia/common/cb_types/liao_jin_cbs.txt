### Liao/(Jurchen) Jin

# Potential Liao-Jin over all of China if the Sixteen Prefectures aren't handed over
khitan_invasion = {
	name = CB_NAME_KHITAN_INVASION
	war_name = WAR_NAME_KHITAN_INVASION
	sprite = 8
	truce_days = 3650
	diplo_view_region = world_china
	is_permanent = no # Only started from script

	allowed_to_target_tributaries = no
	can_ask_to_join_war = yes
	can_call_allies = yes
	allow_whitepeace = no # You win, or you lose

	can_use = {
		always = no # Should only be triggered in script
	}
	
	is_valid = {
		ROOT = {
			government = chinese_imperial_government
		}
	}

	on_success = {
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
		
		attacker = {
			participation_scaled_prestige = 1000
		}
		
		# 1: Usurp all vassals
		defender = {
			any_vassal = {
				limit = {
					higher_tier_than = baron
				}
				
				set_defacto_liege = attacker
			}
		}
		
		# 2: Destroy any emperor titles held by FROM and vassalize and imprison FROM
		defender = {
			any_demesne_title = {
				limit  = {
					tier = emperor
				}
				
				clear_all_chinese_dynasty_flags_and_update_global_flags_effect = yes
				
				unsafe_destroy_landed_title = THIS
			}
			
			set_defacto_liege = attacker
			
			imprison = attacker
			
			add_trait = former_eoc_trait
		}
		
		# 3: Prestige for allies
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
		}
		
		# Prestige loss for the defender
		defender = {
			prestige = -1000
		}
	}

	on_reverse_demand = {
		attacker = {
			transfer_scaled_wealth = {
				to = FROM
				value = 2.0
			}
			
			prestige = -1000
		}
		
		defender = {
			participation_scaled_prestige = 2000
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 2000
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
	}
	attacker_ai_victory_worth = {
		factor = -1
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

# (Jurchen) Jin puppet defection CB
puppet_independence = {
	name = CB_NAME_PUPPET_INDEPENDENCE
	war_name = WAR_NAME_PUPPET_INDEPENDENCE
	sprite = 8
	truce_days = 3650
	
	diplo_view_region = world_china
	is_permanent = no # Only started from script

	allowed_to_target_tributaries = no
	can_ask_to_join_war = yes
	can_call_allies = yes
	allow_whitepeace = no # You win, or you lose

	can_use = {
		always = no # Should only be triggered in script
	}
	
	on_success = {
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
		
		attacker = {
			prestige = 1000
			participation_scaled_prestige = 1000
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
		}
		
		defender = {
			prestige = -1000
			participation_scaled_prestige = -1000
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = -1000
			}
		}
		
		defender = {
			# Take all Jurchen land in China
			any_demesne_title = {
				limit = {
					OR = {
						tier = count
						tier = baron
					}
					capital_scope = {
						region = world_china
					}
				}
				
				usurp_title_plus_barony_if_unlanded = attacker
			}
			
			any_realm_lord = {
				limit = {
					any_demesne_title = {
						OR = {
							tier = count
							tier = baron
						}
						capital_scope = {
							region = world_china
						}
					}
				}
				
				if = {
					limit = {
						NOR = {
							dynasty = defender
							any_demesne_title = {
								OR = {
									tier = count
									tier = baron
								}
								NOT = {
									capital_scope = {
										region = world_china
									}
								}
							}
						}
					}
					
					# Any vassal holding land in China that's NOT of FROM's dynasty becomes a vassal of ROOT if not also holding land elsewhere
					set_defacto_liege = attacker
				}
				
				else = {
					any_demesne_title = {
						limit = {
							OR = {
								tier = count
								tier = baron
							}
							capital_scope = {
								region = world_china
							}
						}
						
						# All other vassals lose any land they hold in China
						usurp_title_plus_barony_if_unlanded = attacker
					}
				}
			}
			
			# If at war with China using the relevant CB for the historical war, end that war with reverse_demand
			if = {
				limit = {
					any_war = {
						attacker = {
							character = defender
						}
						
						using_cb = jurchen_invasion_2
					}
				}
				
				any_war = {
					limit = {
						attacker = {
							character = defender
						}
						
						using_cb = jurchen_invasion_2
					}
					
					end_war = reverse_demand
				}
			}
		}
		
		attacker = {
			e_china = {
				add_weak_pressed_claim = attacker # You are an actual pretender now, aren't you?
			}
		}
	}
	
	on_reverse_demand = {
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
		
		attacker = {
			prestige = -1000
			
			participation_scaled_prestige = -1000
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = {
				participation_scaled_prestige = -1000
			}
		}
		
		defender = {
			prestige = 1000
			
			participation_scaled_prestige = 1000
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
		}
		
		attacker = {
			any_vassal = {
				set_defacto_liege = defender
			}
			
			any_demesne_title = {
				limit = {
					tier = emperor
				}
				unsafe_destroy_landed_title = THIS
			}
			
			set_defacto_liege = defender
			
			imprison = defender
			
			add_trait = former_eoc_trait
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}
