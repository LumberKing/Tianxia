mandate_of_heaven_uprising = {
	name = CB_NAME_MANDATE_OF_HEAVEN_UPRISING
	war_name = WAR_NAME_MANDATE_OF_HEAVEN_UPRISING
	sprite = 16
	truce_days = 365

	hostile_against_others = yes
	check_de_jure_tier = KING # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character
	press_claim = yes
	can_ask_to_join_war = no
	is_independence = yes

	can_use = {
		# Attacker
		ROOT = { has_character_flag = mandate_of_heaven_uprising }
	}

	can_use_title = {
		tier = KING
	}

	on_add = {
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success = {
		attacker = {
			show_scope_change = no

			prestige = 1000
			occupy_minors_of_occupied_settlements = defender

			gain_all_occupied_titles = {
				who = defender
				type = revolt
			}
		}

		defender = {
			show_scope_change = no

			prestige = -500
			
			hidden_tooltip = {
				if = {
					limit = {
						government = chinese_imperial_government
					}
					mandate_penalty_massive_effect = yes
				}
			}
		}
	}

	on_success_title = {
		custom_tooltip = {
			text = other_invasion_succ_tip

			attacker = {
				vassalize_or_take_under_title = {
					title = PREV
					enemy = defender
					is_revolt = yes
					type = revolt
				}

				if = { # Usurp or create kingdom title if enough land was taken
					limit = {
						num_of_count_titles >= 3

						PREV = {
							OR = {
								has_holder = no
								holder = defender
								holder_scope = {
									liege = {
										character = defender
									}
								}
							}
						}
					}
					
					PREV = {
						holder_scope = {
							any_vassal = {
								limit = {
									tier = duke
								}
								set_defacto_liege = defender
							}
						}
					}

					usurp_title = {
						target = PREV
						type = revolt
					}
				}
				
				hidden_tooltip = {
					if = {
						limit = {
							any_demesne_title = {
								tier = baron
								holding_type = castle
							}
						}
						
						# Should we have become a patrician by accident, fix that!
						any_demesne_title = {
							limit = {
								tier = baron
								is_titular = yes
							}
							
							unsafe_destroy_landed_title = THIS
						}
						
						set_correct_feudal_government_type_effect = yes # Fix stupid MRs!
					}
				}
				
				hidden_tooltip = {
					primary_title = {
						add_law = ze_administration_laws_2
						add_law = succ_primogeniture
					}
				}
			}
		}

		hidden_effect = {
			check_if_reconquista_finished_effect = yes

			defender = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail_title = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = mandate_of_heaven_uprising

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}

				remove_trait = inspiring_leader
			}
		}

		defender = {
			show_scope_change = no

			prestige = 100

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
			
			hidden_tooltip = {
				if = {
					limit = {
						government = chinese_imperial_government
					}
					mandate_boost_large_effect = yes
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = nationalist_rebel

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}

				remove_trait = inspiring_leader
			}
		}

		defender = {
			show_scope_change = no

			prestige = 200

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}

				add_character_modifier = {
					name = crushed_major_revolt
					years = 10
				}
			}
			
			hidden_tooltip = {
				if = {
					limit = {
						government = chinese_imperial_government
					}
					mandate_boost_massive_effect = yes
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = 100
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}
