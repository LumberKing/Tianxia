obj_become_council_member = {
	type = character

	is_job_ambition = yes
	expectation_of_liege = yes

	potential = {
		has_dlc = "Conclave"
		is_landed = yes
		is_adult = yes
		independent = no
		is_councillor = no
		prisoner = no
		is_actually_incapable_trigger = no
		is_inaccessible_trigger = no
		NOT = { is_married = liege }

		OR = {
			can_hold_title = job_chancellor
			can_hold_title = job_marshal
			can_hold_title = job_treasurer
			can_hold_title = job_spymaster
			can_hold_title = job_spiritual
		}
	}

	success = {
		is_councillor = yes
		#NOT = {
		#	is_council_eunuch_trigger = yes # TODO
		#}
	}

	effect = {
		opinion = {
			name = opinion_ambition_fulfilled
			who = liege
			years = 5
		}
	}

	abort = {
		OR = {
			independent = yes
			is_landed = no
			is_actually_incapable_trigger = yes

			NOR = {
				can_hold_title = job_chancellor
				can_hold_title = job_marshal
				can_hold_title = job_treasurer
				can_hold_title = job_spymaster
				can_hold_title = job_spiritual
			}
		}
	}

	chance = {
		factor = 100

		mult_modifier = {
			factor = 0.5
			trait = content
		}

		mult_modifier = {
			factor = 3.0
			trait = ambitious
		}

		mult_modifier = {
			factor = 0.5
			trait = shy
		}

		mult_modifier = {
			factor = 1.5
			trait = gregarious
		}

		mult_modifier = {
			factor = 1.5
			trait = proud
		}

		mult_modifier = {
			factor = 0.5
			trait = humble
		}

		mult_modifier = {
			factor = 1.25
			liege = { dynasty = PREV }
		}

		mult_modifier = {
			factor = 1.25

			OR = {
				diplomacy >= 10
				stewardship >= 10
				intrigue >= 10
				martial >= 10
				learning >= 10
			}
		}

		mult_modifier = {
			factor = 1.5

			OR = {
				diplomacy >= 16
				stewardship >= 16
				intrigue >= 16
				martial >= 16
				learning >= 16
			}
		}

		mult_modifier = {
			factor = 10
			among_most_powerful_vassals = 5
		}

		mult_modifier = {
			factor = 5
			relative_income_to_liege >= 0.50
		}
	}
}

obj_build_a_war_chest = {
	type = character

	potential = {
		has_dlc = "Conclave"
		is_landed = yes
		higher_real_tier_than = BARON
		is_adult = yes
		is_nomadic = no
		war = no
		is_actually_incapable_trigger = no
		is_inaccessible_trigger = no
		NOT = { has_character_modifier = war_chest_timer }

		trigger_if = {
			limit = { real_tier = COUNT }
			wealth < 300
		}
		trigger_else_if = {
			limit = { real_tier = DUKE }
			wealth < 500
		}
		trigger_else_if = {
			limit = { real_tier = KING }
			wealth < 700
		}
		trigger_else = {
			wealth < 1000
		}
	}

	success = {
		custom_tooltip = {
			text = obj_build_a_war_chest_success_custom_tooltip

			trigger_if = {
				limit = { real_tier = COUNT }
				wealth >= 300
			}
			trigger_else_if = {
				limit = { real_tier = DUKE }
				wealth >= 500
			}
			trigger_else_if = {
				limit = { real_tier = KING }
				wealth >= 700
			}
			trigger_else = {
				wealth >= 1000
			}
		}
	}

	effect = {
		add_character_modifier = {
			name = war_taxes
			years = 5
		}

		add_character_modifier = {
			name = war_chest_timer
			years = 25
			hidden = yes
		}
	}

	abort = {
		OR = {
			is_actually_incapable_trigger = yes
			is_landed = no
			lower_real_tier_than = COUNT
			is_nomadic = yes
		}
	}

	chance = {
		factor = 100

		mult_modifier = {
			factor = 10
			wealth < 0
		}

		mult_modifier = {
			factor = 2
			trait = greedy
		}

		mult_modifier = {
			factor = 2
			trait = diligent
		}

		mult_modifier = {
			factor = 2
			martial >= 10
		}

		mult_modifier = {
			factor = 0.5
			martial < 10
		}

		mult_modifier = {
			factor = 0.5
			independent = no
		}

		mult_modifier = {
			factor = 0.5
			trait = charitable
		}

		mult_modifier = {
			factor = 0.5
			trait = content
		}

		mult_modifier = {
			factor = 0.5
			trait = slothful
		}

		mult_modifier = {
			factor = 0.5

			trigger_if = {
				limit = { real_tier = COUNT }
				wealth >= 200
			}
			trigger_else_if = {
				limit = { real_tier = DUKE }
				wealth >= 350
			}
			trigger_else_if = {
				limit = { real_tier = KING }
				wealth >= 550
			}
			trigger_else_if = {
				limit = { real_tier = EMPEROR }
				wealth >= 850
			}
		}

		mult_modifier = {
			factor = 0.5
			government = theocracy_government
		}

		mult_modifier = {
			factor = 0.5

			lower_real_tier_than = KING
			independent = no
		}
	}
}

obj_groom_an_heir = {
	type = character

	fertility = 0.20

	potential = {
		has_dlc = "Conclave"
		higher_real_tier_than = BARON
		is_adult = yes
		is_actually_incapable_trigger = no
		is_inaccessible_trigger = no
		holy_order = no
		mercenary = no

		trigger_if = {
			limit = { trait = eunuch }
			has_living_children = yes
		}

		trigger_if = {
			limit = { is_female = yes }
			practical_age < 45
		}
		trigger_else = {
			practical_age < 75
		}

		NOT = {
			any_child = {
				dynasty = ROOT
				is_adult = yes
				is_actually_incapable_trigger = no
				has_inheritance_blocker = no

				NOR = {
					trait = bastard
					trait = celibate
				}

				trigger_if = {
					limit = {
						ROOT = { has_law = agnatic_succession }
					}

					is_female = no
				}
				trigger_else_if = {
					limit = {
						ROOT = { has_law = enatic_succession }
					}

					is_female = yes
				}
			}
		}
	}

	success = {
		any_child = {
			dynasty = ROOT
			is_adult = yes
			has_inheritance_blocker = no

			NOT = {
				custom_tooltip = {
					text = bastard_eunuch_celibate_or_incapable_tt # TODO: Make negative version of the tooltip, as the not doesn't appear in the tooltip

					OR = {
						trait = bastard
						trait = eunuch
						trait = celibate
						is_actually_incapable_trigger = yes
					}
				}
			}

			OR = {
				custom_tooltip = { # It's a boy, and you have Agnatic
					text = boy_agnatic_tt

					is_female = no
					ROOT = { has_law = agnatic_succession }
				}

				custom_tooltip = { # It's a girl, and you have Enatic
					text = girl_enatic_tt

					is_female = yes
					ROOT = { has_law = enatic_succession }
				}

				custom_tooltip = {
					text = not_enatic_or_agnatic_tt

					ROOT = {
						NOR = {
							has_law = enatic_succession
							has_law = agnatic_succession
						}
					}
				}
			}
		}
	}

	effect = {
		custom_tooltip = {
			text = opinion_boost_child

			any_child = {
				limit = {
					dynasty = ROOT
					is_adult = yes
					is_actually_incapable_trigger = no
					has_inheritance_blocker = no

					NOR = {
						trait = bastard
						trait = celibate
					}

					trigger_if = {
						limit = { is_female = yes }

						ROOT = {
							NOT = { has_law = agnatic_succession }
						}
					}
					trigger_else = {
						ROOT = {
							NOT = { has_law = enatic_succession }
						}
					}
				}

				opinion = {
					name = opinion_family_love
					who = ROOT
					years = 20
				}
			}
		}
		
		if = {
			limit = {
				OR = {
					religion = confucian
					has_religion_feature = religion_feature_family_focus
					has_religion_feature = religion_feature_batak
					has_religion_feature = religion_feature_miao
					has_religion_feature = religion_feature_sundanese
					has_religion_feature = religion_feature_viet
					has_religion_feature = religion_feature_yi
					has_religion_feature = religion_feature_zhuang
				}
			}
			if = {
				limit = {
					OR = {
						real_tier = emperor
						controls_religion = yes
					}
				}
				piety = 500
			}
			else_if = {
				limit = {
					real_tier = king
				}
				piety = 250
			}
			else_if = {
				limit = {
					real_tier = duke
				}
				piety = 100
			}
			else = {
				piety = 50
			}
		}
	}

	abort = {
		OR = {
			is_actually_incapable_trigger = yes
			lower_real_tier_than = COUNT
			holy_order = yes
			mercenary = yes

			trigger_if = {
				limit = { trait = eunuch }
				has_living_children = no
			}

			trigger_if = {
				limit = { is_female = yes }
				practical_age >= 45
			}
			trigger_else = {
				practical_age >= 75
			}

			AND = {
				is_feudal = no
				is_tribal = no
				is_nomadic = no
				is_patrician = no
			}
		}
	}

	chance = {
		factor = 100

		trigger = {
			is_theocracy = no
			NOT = { trait = celibate }
		}

		mult_modifier = {
			factor = 10
			has_focus = focus_family
		}

		mult_modifier = {
			factor = 5
			trait = bastard
		}

		mult_modifier = {
			factor = 5
			practical_age >= 30
		}

		mult_modifier = {
			factor = 5
			practical_age >= 45
		}

		mult_modifier = {
			factor = 5
			is_nomadic = yes
		}

		mult_modifier = {
			factor = 2
			is_smart_trigger = yes
		}

		mult_modifier = {
			factor = 0.5
			trait = shy
		}

		mult_modifier = {
			factor = 0.5
			trait = chaste
		}
		
		mult_modifier = {
			factor = 5
			OR = {
				religion = confucian
				has_religion_feature = religion_feature_family_focus
				has_religion_feature = religion_feature_batak
				has_religion_feature = religion_feature_miao
				has_religion_feature = religion_feature_sundanese
				has_religion_feature = religion_feature_viet
				has_religion_feature = religion_feature_zhuang
			}
		}
	}
