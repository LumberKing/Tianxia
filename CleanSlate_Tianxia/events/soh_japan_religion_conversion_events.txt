# Religion conversion chance in Japan
character_event = {
	id = soh.7601
	hide_window = yes
	
	only_playable = yes
	
	is_triggered_only = yes # on_decade_pulse
	
	religion_group = indian_group
	
	trigger = {
		religions_are_standard_trigger = yes
		
		OR = {
			has_japanese_religion_trigger = yes
			AND = {
				religion = buddhist
				culture_group = japonic
			}
		}
		
		any_courtier = {
			is_landed = no
			is_adult = yes
			is_inaccessible_or_incapable_trigger = no
			is_ascetic_trigger = no
			
			OR = {
				has_japanese_religion_trigger = yes
				AND = {
					religion = buddhist
					culture_group = japonic
				}
			}
			
			NAND = {
				religion = shinto
				has_secret_religion = no
				any_dynasty_member = {
					is_the_tenno_trigger = yes
				}
			}
			
			NOR = {
				AND = {
					OR = {
						trait = zealous
						trait = proud
					}
					NOR = {
						has_secret_japanese_religion_trigger = yes
						AND = {
							secret_religion = buddhist
							culture_group = japonic
						}
					}
				}
				has_character_flag = japanese_religion_switch
				has_character_flag = japan_converted
				has_character_flag = no_japaneese_religion_switch
				trait = japanese_monastic_trait
			}
		}
	}
	
	immediate = {
		any_courtier = {
			limit = {
				is_landed = no
				is_adult = yes
				is_inaccessible_or_incapable_trigger = no
				is_ascetic_trigger = no
				
				OR = {
					has_japanese_religion_trigger = yes
					AND = {
						religion = buddhist
						culture_group = japonic
					}
				}
				
				NAND = {
					religion = shinto
					has_secret_religion = no
					any_dynasty_member = {
						is_the_tenno_trigger = yes
					}
				}
				
				NOR = {
					AND = {
						OR = {
							trait = zealous
							trait = proud
						}
						NOR = {
							has_secret_japanese_religion_trigger = yes
							AND = {
								secret_religion = buddhist
								culture_group = japonic
							}
						}
					}
					has_character_flag = japanese_religion_switch
					has_character_flag = japan_converted
					has_character_flag = no_japaneese_religion_switch
					trait = japanese_monastic_trait
				}
			}
			
			random_list = {
				90 = {
					mult_modifier = {
						factor = 2
						trait = content
					}
					
					random_list = {
						50 = {
							# Do nothing
						}
						
						50 = {
							# Make ineligible for future events; reduce spam!
							set_character_flag = no_japaneese_religion_switch
						}
					}
				}
				
				10 = {
					mult_modifier = {
						factor = 2
						has_secret_religion = yes
					}
					
					set_character_flag = japanese_religion_switch
					
					random_list = {
						25 = {
							trigger = {
								NOT = {
									religion = shinto
								}
							}
							
							mult_modifier = {
								factor = 100
								true_religion = shinto
							}
							
							religion = shinto
						}
						
						25 = {
							trigger = {
								NOT = {
									religion = buddhist
								}
							}
							
							mult_modifier = {
								factor = 100
								true_religion = buddhist
							}
							
							religion = buddhist
							
							if = {
								limit = {
									host = {
										religion = buddhist
										OR = {
											trait = mahayana_buddhist
											trait = vajrayana_buddhist
											trait = theravada_buddhist
										}
									}
								}
								
								host = {
									trigger_switch = {
										on_trigger = trait
										mahayana_buddhist  = { PREV = { add_trait = mahayana_buddhist } }
										vajrayana_buddhist = { PREV = { add_trait = vajrayana_buddhist } }
										theravada_buddhist = { PREV = { add_trait = theravada_buddhist } }
									}
								}
							}
							
							else = {
								add_trait = mahayana_buddhist
							}
						}
						
						25 = {
							trigger = {
								NOT = {
									religion = shugendo
								}
							}
							
							mult_modifier = {
								factor = 100
								true_religion = shugendo
							}
							
							religion = shugendo
						}
						
						25 = {
							trigger = {
								NOT = {
									religion = koshin
								}
							}
							
							mult_modifier = {
								factor = 100
								true_religion = koshin
							}
							
							religion = koshin
						}
					}
				}
			}
		}
	}
}

### Provinces convert

# Shinto -> anything
province_event = {
	id = soh.7602
	desc = EVTDESC_SOH_7602
	picture = GFX_evt_burning_house
	border = GFX_event_normal_frame_religion
	
	trigger = {
		religion = shinto
		
		religions_are_standard_trigger = yes
	}
	
	mean_time_to_happen = {
		months = 1200
		
		mult_modifier = {
			factor = 2.0
			religion_authority >= 0.6
		}
		
		mult_modifier = {
			factor = 2.0
			religion_authority >= 0.8
		}
		
		mult_modifier = {
			factor = 2
			has_game_rule = {
				name = religious_conversion
				value = slower
			}
		}
		
		mult_modifier = {
			factor = 0.75
			NOT = { religion_authority > 0.2 }
		}
		
		mult_modifier = {
			factor = 0.75
			NOT = { religion_authority > 0.4 }
		}
	}
	
	immediate = {
		random_list = {
			33 = {
				trigger = {
					culture_group = japonic
				}
				
				religion = buddhist
			}
			
			34 = {
				religion = shugendo
			}
			
			34 = {
				religion = koshin
			}
		}
		
		add_province_modifier = {
			name = religious_unrest
			duration = 730
		}
	}
	
	option = {
		name = {
			trigger = {
				owner = {
					NOR = {
						religion = shinto
						religion = ROOT
					}
				}
			}
			
			text = EVTOPTA_SOH_7602
		}
		
		name = {
			trigger = {
				owner = {
					religion = ROOT
				}
			}
			
			text = EVTOPTB_SOH_7602
		}
		
		name = {
			trigger = {
				owner = {
					religion = shinto
				}
			}
			
			text = EVTOPTC_SOH_7602
		}
		
		if = {
			limit = { 
				owner = {
					religion = shinto
				}
			}
			
			ruler = { piety = -10 }
		}
		
		# News spread
		
		hidden_tooltip = {
			owner = {
				top_liege = {
					if = {
						limit = {
							NOT = { character = PREV }
						}
						
						character_event = { id = soh.7603 days = 2 }
					}
					
					any_vassal = {
						limit = {
							OR = {
								religion = shinto
								religion = ROOT
							}
						}
						
						character_event = { id = soh.7603 days = 2 }
					}
				}
			}
		}
	}
}
