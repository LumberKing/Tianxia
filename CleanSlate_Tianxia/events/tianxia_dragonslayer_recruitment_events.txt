# Sir Genericus, at your service!
narrative_event = {
	id = tianxia.102
	title = EVTTITLE_TIANXIA_102
	desc = EVTDESC_TIANXIA_102
	picture = GFX_evt_rider_adventure
	border = GFX_event_narrative_frame_war
	
	is_triggered_only = yes
	
	immediate = {
		any_province = {
			limit = {
				has_province_flag = dragons_lair_province
			}
			
			random_province = {
				limit = {
					has_owner = yes
					distance = {
						where = PREV
						value < 250
					}
				}
				save_event_target_as = dragonslayer_province
			}
		}
		
		create_character = {
			age = 35
			female = no
			dynasty = none
			culture = event_target:dragonslayer_province
			religion = event_target:dragonslayer_province
			
			attributes = {
				diplomacy = 5
				martial = 7
				stewardship = 4
				intrigue = 2
				learning = 3
			}
			
			random_traits = yes
			trait = skilled_tactician
			trait = brave
		}
		
		new_character = {
			save_event_target_as = new_dragonslayer
			set_character_flag = ai_flag_refuse_concubinage
			set_character_flag = ai_flag_refuse_marriage
			set_character_flag = no_court_invites
			set_character_flag = dragonslayer_sir_genericus
			
			add_one_duel_experience_modifier = yes
			add_one_duel_experience_modifier = yes
			add_one_duel_experience_modifier = yes
			add_one_duel_experience_modifier = yes
			add_one_duel_experience_modifier = yes
		}
	}
	
	option = {
		name = EVTOPTA_TIANXIA_102 # Welcome aboard!
		
		custom_tooltip = {
			text = DRAGONSLAYER_JOINS_THE_PARTY
			
			event_target:new_dragonslayer = {
				add_trait = hunting_dragons
				set_character_flag = do_not_disturb
				
				opinion = {
					who = ROOT
					modifier = hunting_dragons_together
				}
			}
		}
	}
	
	option = {
		trigger = {
			ai = no
		}
		
		name = EVTOPTB_TIANXIA_102 # Get lost!
		
		event_target:new_dragonslayer = {
			leave_court_and_die_effect = yes
		}
	}
}
