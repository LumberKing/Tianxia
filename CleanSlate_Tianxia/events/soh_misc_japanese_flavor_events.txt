# Hidden first event
character_event = {
	id = soh.7501
	hide_window = yes
	
	is_triggered_only = yes
	
	only_playable = yes
	
	trigger = {
		owns = event_target:pulse_province
		
		NOT = {
			has_global_flag = get_medieval
		}
		
		year = 1337 # England and France are at it... not that that narrows it down too much...
		
		NOT = {
			has_game_rule = {
				name = supernatural_events # Sufficiently advanced technology and all that...
				value = off
			}
		}
	}
	
	immediate = {
		set_global_flag = get_medieval
		
		2485 = {
			province_event = { id = soh.7502 days = 3650 random = 7300 }
		}
	}
}

# And hidden second event
province_event = {
	id = soh.7502
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				has_game_rule = {
					name = supernatural_events # Sufficiently advanced technology and all that...
					value = unrestricted
				}
			}
			
			any_playable_ruler = {
				limit = {
					is_inaccessible_or_incapable_trigger = no
					any_realm_province = {
						region = world_greater_japan
					}
				}
				
				character_event = { id = soh.7503 } # Actual event
			}
		}
		
		else = {
			any_player = {
				limit = {
					is_inaccessible_or_incapable_trigger = no
					any_realm_province = {
						region = world_greater_japan
					}
				}
				
				character_event = { id = soh.7503 } # Actual event
			}
		}
	}
}
