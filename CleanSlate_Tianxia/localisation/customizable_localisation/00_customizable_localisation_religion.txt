	use_first_valid = yes

			}
		}
	}
	text = {
		localisation_key = String_spahi
		trigger = {
			culture_group = iranian
		}

		localisation_key = String_sainik
		trigger = {
			OR = {
				culture_group = indo_aryan_group

			}
		}
	}
	text = {
		localisation_key = String_kheshig
		trigger = {
			culture_group = altaic
		}
	}
	text = {
		localisation_key = String_man_at_arm
