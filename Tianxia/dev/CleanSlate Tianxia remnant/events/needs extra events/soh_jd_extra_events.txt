# War focus trait gain
character_event = {
	id = jdextra.10
	
	is_triggered_only = yes
	
	has_dlc = "Jade Dragon"
	
	picture = GFX_evt_china_invades
	border = GFX_event_normal_frame_war
	
	desc = EVTDESC_JDEXTRA_10
	
	capable_only = yes
	prisoner = no

	trigger = {
		has_focus = focus_war
		can_have_more_leadership_traits = yes
		OR = {
			culture_group = chinese_group
			has_artifact = art_of_war
			AND = {
				learning = 10
				martial = 15
				any_realm_province = {
					region = world_china
				}
			}
			AND = {
				learning = 15
				martial = 20
				OR = {
					any_realm_province = {
						OR = {
							region = world_far_east
							region = world_tibet
							region = world_indochina
							region = world_mongolia
							region = world_manchuria
						}
						trade_route = silk_road
					}
				}
				has_chinese_or_historical_culture_for_chinese_imperial_trigger = yes
			}
			AND = {
				learning = 20
				martial = 25
				OR = {
					any_realm_province = {
						OR = {
							region = world_far_east
							region = world_tibet
							region = world_indochina
							region = world_mongolia
							region = world_manchuria
						}
						trade_route = silk_road
					}

			}
		}

			has_character_flag = chinese_warfare

		}
		modifier = {
			factor = 0.1
			NOT = { martial = 5 }
		}
		modifier = {
			factor = 0.5
			NOT = { martial = 8 }
		}
		modifier = {
			factor = 2.0
			NOT = { martial = 12 }
		}
		modifier = {
			factor = 2.0
			has_character_modifier = adept_of_strategy
		}
		modifier = {
			factor = 3.0
			trait = strategist
		}

			factor = 0.25
			leader_traits = 1
		}
		modifier = {
			factor = 0.01
			is_nomadic = yes
		}
		modifier = {
			factor = 0.1
			is_tribal = yes
		}
	}
	
	immediate = {
		set_character_flag = chinese_warfare
	}
	
	option = {

			factor = 100
		}
