# Strange banyan tree - investigate?
character_event = {
	id = artifacts.12
	desc = EVTDESC_ARTIFACTS_12
	picture = GFX_evt_emissary
    border = GFX_event_normal_frame_diplomacy
	
	is_triggered_only = yes
	
	immediate = {
		random_realm_province = {
			limit = {
				region = world_indochina
			}
			save_event_target_as = artifact_province
		}
	}
	
	option = {
		name = EVTOPTA_ARTIFACTS_12 # Let us find this tree!
		
		hidden_tooltip = {
			random_list = {
				33 = {
					character_event = { id = artifacts.13 days = 30 random = 30 } # Tree burned down
				}
				33 = {
					character_event = { id = artifacts.14 days = 30 random = 30 } # Tree chopped down
				}
				34 = {
					narrative_event = { id = artifacts.15 days = 30 random = 30 } # Tree exists
				}
			}
		}
	}
	
	option = {
		name = EVTOPTB_ARTIFACTS_12 # I have no time for this!
		
		prestige = -10
		
		ai_chance = {
			factor = 0
		}
	}
}

# Both halves found - get Thuan Thien (also possible to trigger from a decision)

	id = artifacts.20
	desc = EVTDESC_ARTIFACTS_20
	picture = GFX_evt_china_expansionist
    border = GFX_event_normal_frame_war
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_ARTIFACTS_20 # Great!
		
		piety = 100
		prestige = 100
		
		any_artifact = {
			limit = {
				has_artifact_flag = thuan_thien
			}
			unsafe_destroy_artifact = yes
		}
		
		add_artifact = thuan_thien
	}
}
