	title = EVTTITLE_SOH_8502

	picture = GFX_evt_treasure_fleet
	window = EventWindowOffmap
	background = GFX_event_window_news_from_china
	
	is_triggered_only = yes
	
	option = {
		trigger = {
			trait = treasure_fleet_voyage
		}
		
		name = EVTOPTA_SOH_8502 # Time to go ashore!
	}
	
	option = {
		trigger = {
			NOT = {
				trait = treasure_fleet_voyage
			}
			top_liege = {
				has_landed_title = e_china
			}
		}
		
		name = EVTOPTB_SOH_8502 # I suppose they'll resupply here...
	}
	
	option = {
		trigger = {
			NOR = {
				trait = treasure_fleet_voyage
				top_liege = {
					has_landed_title = e_china
				}
			}
			top_liege = {
				government = chinese_imperial_government
			}
		}
		
		name = EVTOPTC_SOH_8502 # They are not welcome here!
	}
	
	option = {
		trigger = {
			NOR = {
				trait = treasure_fleet_voyage

					has_landed_title = e_china

			}
