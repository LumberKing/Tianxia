				holder_scope = {

						}
					}
					# The AI should not do this if it shares the dynasty of the EoC
					OR = {
						ROOT = {

						}

						}
					}
				}
			}
			NOT = {

		}

			unsafe_destroy_artifact = deed_to_the_dragon_throne

				add_pressed_claim = ROOT
