	convert_to_japanese = {
		only_playable = yes
		
		filter = self
		ai_target_filter = self
		ai_check_interval = 60
	
		potential = {
			is_playable = yes
			OR = {
				culture = yamato
				culture = emishi
			}
			capital_scope = {
				region = world_japan
			}
			any_realm_province = {
				culture = japanese
			}
			NOT = {
				has_alternate_start_parameter = { key = culture_names value = random }
			}
		}
		allow = {
			OR = {
				culture = yamato
				culture = emishi
			}
		}
		effect = {
			culture = japanese

				limit = {
					OR = {
						culture = yamato
						culture = emishi
					}
					employer = { character = ROOT }
					dynasty = ROOT
				}
				culture = japanese
			}
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 1

				factor = 0.5 # Slow it down a bit
			}
			
			modifier = {
				factor = 0.01
				capital_scope = {
					NOT = { culture = japanese }
				}
			}
			modifier = {
				factor = 2.0
				liege = { culture = japanese }
			}
			modifier = {
				factor = 1.1
				num_culture_realm_provs = {
					value = 2
					culture = japanese
				}
			}
			modifier = {
