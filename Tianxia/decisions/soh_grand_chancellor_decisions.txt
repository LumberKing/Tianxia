			NOT = {
				any_owned_bloodline = {
					has_bloodline_flag = grand_chancellor_bloodline
					bloodline_is_active_for = PREV
				}
			}
			OR = {
				prestige = 4000
				AND = {
					liege = {
						has_regent = yes
					}
					prestige = 1000
				}
			}
			liege = {
				primary_title = {
					has_law = law_voting_power_1
				}
				OR = {
					# Majority support from the council
					AND = {
						any_voter = {
							count = 4
							is_council_eunuch_trigger = no
							has_opinion_modifier = {
								who = ROOT
								modifier = supporting_as_grand_chancellor
							}
						}
					}
					AND = {
						tier = emperor
						any_voter = {
							count = 7
							is_council_eunuch_trigger = yes
							has_opinion_modifier = {
								who = ROOT
								modifier = supporting_as_grand_chancellor
							}
						}
					}
				}
			}
		}
		effect = {
			custom_tooltip = {
				text = create_grand_chancellor_bloodline_effect
				character_event = { id = grandchancellor.14 } # Creathe the bloodline
			}
		}
	}
