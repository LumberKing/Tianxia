		icon_religion = taoist

				text = chinese_enough_has_not_converted
				NOT = { has_character_flag = chinese_enough_converted }
			}
		}
		effect = {
			if = {
				limit = { higher_tier_than = BARON }
				religion_authority = {
					modifier = ruler_converted_from
				}

			if = {
				limit = { higher_tier_than = BARON }

		}
		ai_will_do = {
			factor = 0
