# on_startup event

	hide_window = yes

		}

character_event = {
	id = soh.3503
	hide_window = yes
	
	is_triggered_only = yes
	
	religion = shinto
	only_playable = yes
	
	trigger = {
		controls_religion = yes
		current_heir = {
			NOT = {
				dynasty = ROOT
			}
		}
		NOT = {
			has_global_flag = end_of_the_imperial_family
		}
	}
	
	immediate = {
		set_global_flag = end_of_the_imperial_family # Don't do this more than once...
		
		k_chrysanthemum_throne = {
			unsafe_destroy_landed_title = THIS
		}
		any_demesne_title = {
			limit = {
				tier = emperor
			}
			unsafe_destroy_landed_title = THIS
		}
		any_demesne_title = {
			limit = {
				tier = king
			}
			unsafe_destroy_landed_title = THIS
		}
		any_demesne_title = {
			limit = {
				tier = duke
			}
			unsafe_destroy_landed_title = THIS
		}
		
		# Hurt Shinto MA

			modifier = imperial_family_line_ended
			years = 100
		}

			modifier = amaterasu_defied

		}

	}
}

##### Japanese government trait handling

# Character with Japanese Feudal dies - all unlanded children with the same gender law as their primary that are eligible inherit the trait
character_event = {
	id = soh.3505
	hide_window = yes
	
	is_triggered_only = yes
	
	only_playable = yes
	
	trigger = {
		trait = japanese_feudal_trait
	}
	
	immediate = {
		any_child = {
			limit = {
				can_have_japanese_bureaucracy_and_feudal_trigger = yes
				is_landed = no
				NOT = {
					trait = japanese_monastic_trait
				}
			}
			if = { # Only men
				limit = {
					ROOT = {
						primary_title = {
							OR = {
								has_law = agnatic_succession
								has_law = cognatic_succession # Ag-Cog
							}
						}
					}
					is_female = no
				}
				add_trait = japanese_feudal_trait
			}
			else_if = { # Only women
				limit = {
					ROOT = {
						primary_title = {
							OR = {
								has_law = enatic_succession
								has_law = enatic_cognatic_succession # En-Cog
							}
						}
					}
					is_female = yes
				}
				add_trait = japanese_feudal_trait
			}
			else = {
				add_trait = japanese_feudal_trait
			}
		}
	}
}

							}
						}
					}
