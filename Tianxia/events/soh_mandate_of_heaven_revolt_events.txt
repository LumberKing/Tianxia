### tOG revolts, modelled after liberation revolts
# Liberation-style revolt against a low MoH realm
# Triggered from "on_rebel_revolt"
province_event = {
	id = soh.5901
	desc = EVTDESC_SOH_5901
	picture = GFX_evt_china_unrest
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes

	trigger = {
		has_dlc = "The Old Gods"
		can_have_chinese_imperial_trigger = yes

		any_province_holding = {
			NOT = { holding_type = nomad }
		}
		
		owner = {
			top_liege = {
				government = chinese_imperial_government
				OR = {
					AND = {
						has_character_modifier = mandate_of_heaven_2 # China is held to a higher standard
						has_landed_title = e_china
					}
					has_character_modifier = mandate_of_heaven_1
				}
			}
		}
		
		county = {
			OR = {
				region = world_china
				owner = {
					top_liege = {
						primary_title = {
							de_jure_vassal_or_below = PREVPREV
						}
					}
				}
			}
		}
		NOT = { # Not if there is already an ongoing MoH revolt for the title
			owner = {
				top_liege = {
					war = yes
					any_war = {
						defender = { character = PREV }
						using_cb = mandate_of_heaven_uprising
						thirdparty_title_scope = {
							ROOT = {
								kingdom = {
									title = PREVPREV
								}
							}
						}
					}
				}
			}
		}
	}
	
	immediate = {
		create_character = {
			random_traits = yes
			dynasty = random
			religion = ROOT
			culture = ROOT
			female = no
			age = 20
			health = 7
			fertility = 0.7
			attributes = {
				diplomacy = 6
				martial = 8
				stewardship = 6
				intrigue = 2
				learning = 6
			}
			trait = just
			trait = brave
			trait = gregarious
			trait = ambitious
			trait = brilliant_strategist
			trait = inspiring_leader
			trait = peasant_leader
		}
		
		new_character = {
			set_character_flag = mandate_of_heaven_uprising
			
			create_title = {
				tier = DUKE
				landless = yes
				temporary = yes
				rebel = yes
				culture = ROOT
				name = "MOH_REVOLT"
				holder = THIS
			}
			
			wealth = 100
			
			spawn_unit = {
				province = ROOT
				home = ROOT
				owner = THIS
				#leader = THIS
				scaled_by_biggest_garrison = 2
				troops = {
					pikemen = { 10 10 }
					heavy_infantry = { 4 4 }
					light_infantry = { 4 4 }
					archers = { 8 8 }
					light_cavalry = { 4 4 }
					knights = { 2 2 }
				}
				attrition = 1.0
			}
			
			spawn_moh_army_effect = yes
			spawn_moh_army_effect = yes
			if = {
				limit = {
					ROOT = {
						owner = {
							top_liege = {
								has_character_modifier = mandate_of_heaven_1
							}
						}
					}
				}
				spawn_moh_army_effect = yes
			}
			
			if = {
				limit = {
					has_game_rule = {
						name = provincial_revolt_strength
						value = powerful
					}
				}
				spawn_moh_army_effect = yes
				wealth = 150
			}
			
			if = {
				limit = {
					has_game_rule = {
						name = provincial_revolt_strength
						value = very_powerful
					}
				}
				spawn_moh_army_effect = yes
				spawn_moh_army_effect = yes
				spawn_moh_army_effect = yes
				wealth = 250
			}
			
			if = {
				limit = {
					has_game_rule = {
						name = provincial_revolt_strength
						value = extremely_powerful
					}
				}
				spawn_moh_army_effect = yes
				spawn_moh_army_effect = yes
				spawn_moh_army_effect = yes
				spawn_moh_army_effect = yes
				spawn_moh_army_effect = yes
				wealth = 350
			}
			
			# DoW on the province top liege
			ROOT = {
				owner = {
					top_liege = {
						reverse_war = {
							target = PREVPREVPREV
							casus_belli = mandate_of_heaven_uprising
							thirdparty_title = ROOT # The county...
							tier = KING # ... but really the de jure kingdom
						}
						reverse_opinion = {
							who = PREVPREVPREV
							modifier = opinion_evil_tyrant
						}
					}
				}
			}
			# TODO: Have the revolter found a dynasty right away?
		}
		
		owner = {
			any_liege = { # Inform the lieges
				character_event = {
					id = soh.5902
				}
			}
		}
	}
	
	option = {
		name = "EVTOPTA_SOH_5901"
	}
}

# Nationalists seize a holding. Fired from 'on_siege_over_winner'.
character_event = {
	id = soh.5903
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		rebel = yes
		FROM = { 
			is_capital = yes # The capital holding
		}
		any_war = {
			attacker = { character = ROOT }
			using_cb = mandate_of_heaven_uprising
		}
	}
	
	immediate = {
		FROM = {
			location = {
				if = {
					limit = {
						NOT = { culture = ROOT }
						kingdom = { culture = ROOT }
					}
					random = {
						chance = 20
						culture = ROOT
					}
				}
				if = {
					limit = {
						culture = ROOT
						NOT = { has_province_modifier = cultural_unrest }
					}
					add_province_modifier = {
						name = cultural_unrest
						duration = 730
					}
				}
			}
		}
	}
}

# Rebels rise up to reinforce an ongoing revolt
# Triggered from "on_rebel_revolt"
province_event = {
	id = soh.5904
	desc = EVTDESC_SOH_5904
	picture = GFX_evt_china_unrest
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes

	trigger = {
		has_dlc = "The Old Gods"

		any_province_holding = {
			NOT = { holding_type = nomad }
		}
		
		# There is already an ongoing MoH revolt for the same kingdom
		owner = {
			top_liege = {
				war = yes
				any_war = {
					defender = { character = PREV }
					using_cb = mandate_of_heaven_uprising
					thirdparty_title_scope = {
						ROOT = {
							kingdom = {
								title = PREVPREV
							}
						}
					}
				}
			}
		}
	}
	
	immediate = {
		add_province_modifier = {
			name = recent_county_uprising
			duration = 1825 # Five years of -100% revolt risk in this county
		}
		owner = {
			top_liege = {
				any_war = {
					limit = {
						defender = { character = PREV }
						using_cb = mandate_of_heaven_uprising
						thirdparty_title_scope = {
							ROOT = {
								kingdom = {
									title = PREVPREV
								}
							}
						}
					}
					attacker = {
						spawn_moh_reinforcements_effect = yes
						if = {
							limit = {
								ROOT = {
									owner = {
										top_liege = {
											has_character_modifier = mandate_of_heaven_1
										}
									}
								}
							}
							spawn_moh_reinforcements_effect = yes
						}
						
						if = {
							limit = {
								has_game_rule = {
									name = provincial_revolt_strength
									value = powerful
								}
							}
							spawn_moh_reinforcements_effect = yes
						}
						
						if = {
							limit = {
								has_game_rule = {
									name = provincial_revolt_strength
									value = very_powerful
								}
							}
							spawn_moh_reinforcements_effect = yes
							spawn_moh_reinforcements_effect = yes
							spawn_moh_reinforcements_effect = yes
						}
						
						if = {
							limit = {
								has_game_rule = {
									name = provincial_revolt_strength
									value = extremely_powerful
								}
							}
							spawn_moh_reinforcements_effect = yes
							spawn_moh_reinforcements_effect = yes
							spawn_moh_reinforcements_effect = yes
							spawn_moh_reinforcements_effect = yes
							spawn_moh_reinforcements_effect = yes
						}
					}
				}
			}
		}
		
		owner = {
			any_liege = { # Inform the lieges
				character_event = {
					id = soh.5905
				}
			}
		}
	}
	
	option = {
		name = "EVTOPTA_SOH_5904"
	}
}
