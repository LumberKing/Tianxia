# Recreation if destroyed (fired from on_society_destroyed)
character_event = {
	id = soh.2506
	
	hide_window = yes
	is_triggered_only = yes # on_society_destroyed
	
	has_dlc = "Holy Fury"
	hide_from = yes
	
	trigger = {
		FROM = { is_society = warriors_of_the_rising_sun }
	}

	immediate = {
		if = {
			limit = {
				is_landed = yes
			}
			
			# Pick someone under you to lead the WotRS...
			
			any_courtier_or_vassal = {
				limit = {
					ai = yes
					is_in_society = no
					is_adult = yes
					block_general_event_trigger = no
					
					has_wotrs_prerequisites = yes
					NOT = { character = ROOT }
				}
				
				score_value = { #Prioritizes selecting some kind of warrior...
					value = 10
					wotrs_recruit_score = yes
					wotrs_revival_selection_score = yes
				}
				
				join_society = warriors_of_the_rising_sun
				
				if = {
					limit = {
						NOT = {
							society = {
								any_society_member = {
									is_society_grandmaster = yes
								}
							}
						}
					}
					set_society_grandmaster = yes # Set character as Grand Master
					add_society_currency_massive_effect = yes
				}
				
				add_society_currency_massive_effect = yes
			}
		}
		
		else = {
			liege = {
				#Pick someone under your *liege* to lead the warrior lodge...
				any_courtier_or_vassal = {
					limit = {
						ai = yes
						is_in_society = no
						is_adult = yes
						block_general_event_trigger = no
						
						has_wotrs_prerequisites = yes
						NOT = { character = ROOT }
					}
					
					score_value = { #Prioritizes selecting some kind of warrior...
						value = 10
						wotrs_recruit_score = yes
						wotrs_revival_selection_score = yes
					}
					
					join_society = warriors_of_the_rising_sun
					
					if = {
						limit = {
							NOT = {
								society = {
									any_society_member = {
										is_society_grandmaster = yes
									}
								}
							}
						}
						
						set_society_grandmaster = yes # Set character as Grand Master
						add_society_currency_massive_effect = yes
					}
				}
			}
		}
		
		##log = "The revive event fired for [From.GetName]."
	}
}

# AI rank up
character_event = {
	id = soh.2507
	hide_window = yes
	
	is_triggered_only = yes
	
	is_in_society = yes
	has_dlc = "Holy Fury"
	ai = yes
	min_age = 16
	
	trigger = { 
		society_member_of = warriors_of_the_rising_sun
		is_inaccessible_or_incapable_trigger = no
		society_can_rank_up = yes
		has_society_currency_minor_trigger = yes
		
		society_rank < 3
	}

	immediate = {
		random_list = {
			50 = {}
			
			50 = {
				trigger = {
					society_rank == 1
					
					OR = {
						has_character_flag = proved_myself_to_mentor
						AND = {
							combat_rating >= 25
							days_in_society > 1825
							NOT = {
								has_any_opinion_modifier = opinion_rising_sun_mentor
							}
						}
					}
				}
				
				mult_modifier = {
					factor = 0.5
					days_in_society < 3650
					NOT = {
						has_character_flag = proved_myself_to_mentor
					}
				}
				
				mult_modifier = {
					factor = 1.5
					combat_rating >= 50
				}
				
				mult_modifier = {
					factor = 1.5
					combat_rating >= 75
				}
				
				mult_modifier = {
					factor = 1.5
					combat_rating >= 100
				}
				
				society_rank_up = 1
				
				if = {
					limit = {
						has_quest = quest_rising_sun_get_mentor
					}
					clr_quest = {
						id = quest_rising_sun_get_mentor
					}
				}
				
				detract_society_currency_medium_effect = yes # at a worst, they are put at 0 currency
			}
			
			50 = {
				trigger = {
					society_rank == 2
					mercenary = no
					
					NOT = {
						society = {
							any_society_member = {
								count = 7
								society_rank == 3
							}
						}
					}
				}
				
				society_rank_up = 1
				detract_society_currency_medium_effect = yes # at a worst, they are put at 0 currency
			}
		}
	}
}
