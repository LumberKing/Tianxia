##### Random accident death

# 1 % chance of dying in a random accident for unlanded characters that plausibly could
character_event = {
	id = tianxia.1401
	hide_window = yes
	
	is_triggered_only = yes # on_yearly_pulse -> random_events -> random_death
	
	min_age = 16
	prisoner = no
	only_capable = yes # Works fine
	
	trigger = {
		is_ruler = no
		is_inaccessible_trigger = no
		NOR = {
			trait = immortal
			trait = kitsune
		}
	}
	
	weight_multiplier = {
		days = 1
		
		mult_modifier = {
			factor = 2
			is_unimportant_trigger = yes
		}
		
		mult_modifier = {
			factor = 0.95
			any_close_relative = {
				real_tier = baron
			}
			NOT = {
				any_close_relative = {
					higher_real_tier_than = baron
				}
			}
		}
		
		mult_modifier = {
			factor = 0.9
			any_close_relative = {
				real_tier = count
			}
			NOT = {
				any_close_relative = {
					higher_real_tier_than = count
				}
			}
		}
		
		mult_modifier = {
			factor = 0.85
			any_close_relative = {
				real_tier = duke
			}
			NOT = {
				any_close_relative = {
					higher_real_tier_than = duke
				}
			}
		}
		
		mult_modifier = {
			factor = 0.8
			any_close_relative = {
				real_tier = king
			}
			NOT = {
				any_close_relative = {
					higher_real_tier_than = king
				}
			}
		}
		
		mult_modifier = {
			factor = 0.75
			any_close_relative = {
				real_tier = emperor
			}
		}
	}
	
	immediate = {
		random_accident_death_effect = yes
	}
}

##### Kinslayer trait fix

# on_startup, convert non-Tribal Kinslayer to Tribal Kinslayer where relevant
character_event = {
	id = tianxia.1403
	hide_window = yes
	
	is_triggered_only = yes # on_startup
	
	only_rulers = yes
	
	trigger = {
		is_tribal = yes
		OR = {
			trait = kinslayer
			trait = familial_kinslayer
			trait = dynastic_kinslayer
		}
	}
	
	immediate = {
		add_trait = tribal_kinslayer
	}
}

##### Seclusion fix, e.g on marriage

# on_host_change, put characters in seclusion if their new host is in seclusion
character_event = {
	id = tianxia.1404
	hide_window = yes
	
	is_triggered_only = yes # on_host_change
	
	prisoner = no # Takes precedence
	
	trigger = {
		is_landed = no
		FROMFROM = {
			has_character_modifier = in_seclusion
		}
	}
	
	immediate = {
		add_character_modifier = {
			name = in_seclusion
			duration = -1
		}
	}
}

##### Fountain of Youth age fix

# on_yearly_pulse, age the character one year
character_event = {
	id = tianxia.1406
	hide_window = yes
	
	is_triggered_only = yes # on_yearly_pulse
	
	has_character_flag = fountain_of_youth_setup
	
	trigger = {
		NOR = {
			trait = immortal
			trait = kitsune
		}
	}
	
	immediate = {
		fountain_of_youth_aging_effect = yes
		
		if = {
			limit = {
				practical_age >= 50
			}
			
			random_list = {
				99 = {
					trigger = {
						OR = {
							AND = {
								practical_age < 65
								health >= 4
							}
							AND = {
								practical_age < 70
								health >= 4.5
							}
							AND = {
								practical_age < 75
								health >= 5
							}
							AND = {
								practical_age < 80
								health >= 5.5
							}
							AND = {
								practical_age < 85
								health >= 6
							}
							AND = {
								practical_age < 90
								health >= 6.5
							}
							AND = {
								practical_age < 95
								health >= 7
							}
							health >= 7.5
						}
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 70
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 75
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 80
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 85
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 90
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 95
					}
					
					mult_modifier = {
						factor = 0.85
						practical_age >= 100
					}
					
					mult_modifier = {
						factor = 0.9
						health < 6.5
					}
					
					mult_modifier = {
						factor = 0.9
						health < 6
					}
					
					mult_modifier = {
						factor = 0.9
						health < 5.5
					}
					
					mult_modifier = {
						factor = 0.9
						health < 5
					}
					
					mult_modifier = {
						factor = 0.9
						health < 4.5
					}
					
					# Survived
				}
				
				1 = {
					character_event = { id = tianxia.1407 days = 1 random = 364 } # Death by natural causes
				}
			}
		}
	}
}
