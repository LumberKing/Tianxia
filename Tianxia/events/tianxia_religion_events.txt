# Confucian branch traits (on conversion, to match RoI.111)
character_event = {
	id = tianxia.1
	desc = {
		trigger = {
			NOT = {
				has_alternate_start_parameter = {
					key = religion_names
					value = random
				}
			}
		}
		text = EVTDESC_tianxia_1
	}
	desc = { #alternate start is currently unsupported in Tianxia
		trigger = {
			has_alternate_start_parameter = {
				key = religion_names
				value = random
			}
		}
		text = EVTDESC_tianxia_1_ALT
	}
	picture = GFX_evt_bishop
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	trigger = {
		religion = confucian
		NOT = { has_character_flag = tianxia_eastern_branch_assigned_separately }
	}
	
	option = {
		name = {
			text = EVTOPTA_tianxia_1
			trigger = {
				NOT = {
					has_alternate_start_parameter = {
						key = religion_names
						value = random
					}
				}
			}
		}
		name = {
			text = EVTOPTA_tianxia_1_ALT
			trigger = {
				has_alternate_start_parameter = {
					key = religion_names
					value = random
				}
			}
		}
		add_trait = traditionalist_confucian
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = {
			text = EVTOPTB_tianxia_1
			trigger = {
				NOT = {
					has_alternate_start_parameter = {
						key = religion_names
						value = random
					}
				}
			}
		}
		name = {
			text = EVTOPTB_tianxia_1_ALT
			trigger = {
				has_alternate_start_parameter = {
					key = religion_names
					value = random
				}
			}
		}
		add_trait = neoconfucian_confucian
		ai_chance = { # TODO: dynamic Neo-Confucianism? The seeds of the movement were laid down by Tang-era philosophers by the 867 start date (and arguably can be traced back to developments that were already present in the 769 start date).
			factor = 25
			modifier = {
				factor = 0
				NOT = { has_global_flag = rise_of_neoconfucianism }
			}
			modifier = {
				factor = 10
				has_global_flag = neoconfucian_orthodoxy_china
				top_liege = { culture_group = chinese_group }
			}
		}
	}
}

	id = tianxia.4
	desc = EVTDESC_tianxia_4
	picture = GFX_evt_bishop

	is_triggered_only = yes
	trigger = { ai = no }
	option = { # Diplomacy
		name = EVTOPTA_tianxia_4

		add_character_modifier = { name = spiritual_diplomacy_modifier years = 10 }
	}

	}
}
