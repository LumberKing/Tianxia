# If the Treasure Fleet hasn't visited Nazca (and isn't currently potentially on its way there), a merchant gets lost and heads east
character_event = {
	id = sri.1
	hide_window = yes
	
	is_triggered_only = yes

	only_playable = yes
	has_dlc = "Sunset Invasion"

	trigger = {
		owns = event_target:pulse_province
		year < 1350
		NOR = {
			2753 = { # Hangzhou
				OR = { # Ongoing Nazca journey
					has_province_flag = treasure_fleet_at_pacific_crossing
					has_province_flag = treasure_fleet_new_world_ongoing
					has_province_flag = treasure_fleet_inca_encounter
				}
			}
			has_global_flag = treasure_fleet_pacific_return
			has_global_flag = china_knows_of_burning_empire
			has_global_flag = lost_merchant_went_east
			has_global_flag = sunrise_invasion_happened
		}
		NOT = {
			has_game_rule = {
				name = inca_invasion
				value = off
			}
		}
		OR = {
			year >= 1200 # Give the Treasure Fleet ample time to do its thing
			AND = {
				has_game_rule = {
					name = inca_invasion
					value = delayed_random
				}
				year >= 1000
			}
		 }
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 2
			year >= 1320
		}

		mult_modifier = {
			factor = 0.25

			has_game_rule = {
				name = inca_invasion
				value = random
			}
		}

		mult_modifier = {
			factor = 0.33

			has_game_rule = {
				name = inca_invasion
				value = delayed_random
			}
		}

		# Prevent the randomized invasions from firing immediately after 50 years of game, see 'immediate' below
		mult_modifier = {
			factor = 0.6 # Equivalent to 40% chance of misfiring

			year < 1300
			total_years_played >= 50
			total_years_played < 100

			OR = {
				has_game_rule = {
					name = inca_invasion
					value = random
				}

				has_game_rule = {
					name = inca_invasion
					value = delayed_random
				}
			}
		}

		mult_modifier = {
			factor = 0.7 # Equivalent to 30% chance of misfiring

			year < 1300
			total_years_played >= 100
			total_years_played < 150

			OR = {
				has_game_rule = {
					name = inca_invasion
					value = random
				}

				has_game_rule = {
					name = inca_invasion
					value = delayed_random
				}
			}
		}

		mult_modifier = {
			factor = 0.85 # Equivalent to 15% chance of misfiring

			year < 1300
			total_years_played >= 150
			total_years_played < 200

			OR = {
				has_game_rule = {
					name = inca_invasion
					value = random
				}

				has_game_rule = {
					name = inca_invasion
					value = delayed_random
				}
			}
		}

		mult_modifier = {
			factor = 0.9625 # Equivalent to 3.75% chance of misfiring

			year < 1300
			total_years_played >= 200

			OR = {
				has_game_rule = {
					name = inca_invasion
					value = random
				}

				has_game_rule = {
					name = inca_invasion
					value = delayed_random
				}
			}
		}
	}
	
	immediate = {
		set_global_flag = lost_merchant_went_east
		any_player = {
			narrative_event = { id = sri.2 } # Inform players
		}
		event_target:pulse_province = {
			province_event = { id = sri.3 days = 730 random = 365 } # Next stage
		}
	}
}

province_event = {
	id = sri.5
	hide_window = yes
	
	has_dlc = "Sunset Invasion"
	
	is_triggered_only = yes
	
	immediate = {
		set_global_flag = sunrise_invasion_prelude
		
		# Activate titles and do some stuff
		
		activate_title = { title = e_inca_empire status = yes }
		
		create_random_soldier = {
			female = no
			dynasty = 1000050000 # Yupanqui
			random_traits = yes
			culture = quechua
			religion = inca_pagan
		}
		
		new_character = {
			if = {
				limit = {
					has_game_rule = {
						name = invader_ma_boost
						value = enabled
					}
				}
				
				religion_authority = {
					name = invader_ma_boost
				}
			}
			
			grant_title = e_inca_empire
			
			set_government_type = feudal_government
			
			e_inca_empire = {
				add_law = succ_primogeniture
				add_law = agnatic_succession
				
				if = {
					limit = {
						has_dlc = "Zeus"
					}
					add_law = revoke_title_law_2
				}
				
				if = {
					limit = {
						NOT = { has_dlc = "Zeus" }
					}
					add_law = crown_authority_1
				}
			}
			
			set_defacto_liege = THIS
			wealth = 10000
		}
		
		any_player = {
			narrative_event = { id = sri.6 } # Inform players
		}
		province_event = { id = sri.7 days = 180 random = 180 }
		set_variable = {
			which = sunrise_counter
			value = 0
		}
	}
}
