# Make the AI join
character_event = {
	id = soh.804
	
	hide_window = yes
	is_triggered_only = yes # on_five_year_pulse
	
	only_independent = yes

	trigger = {
		is_alive = yes
		NOT = {
			has_global_flag = hwarang_disbanded
		}
	}

	immediate = {
		any_realm_lord = {
			count = 2
			
			limit = {
				ai = yes
				is_in_society = no
				has_secret_religion = no
				age >= 12
				block_general_event_trigger = no
				higher_real_tier_than = BARON
				
				NOT = {
					has_character_modifier = not_interested_in_hwarang

			score_value = { #Prioritizes selecting some kind of warrior...
				value = 10
				warrior_recruit_score = yes # TIANXIATODO: Make more appropriate?
			}

				limit = {
					OR = {
						is_female = no
						has_religion_feature = religion_equal
						has_religion_feature = religion_matriarchal
					}
				}
				
				if = {
					limit = {
						OR = {
							is_adult = yes

					}
				}

### Maintenance: Hwarang spawns if it died (fired from on_society_destroyed)
# Root = Character
# From = Society
character_event = {
	id = soh.807
	
	hide_window = yes
	is_triggered_only = yes # on_society_destroyed
	
	trigger = {
		FROM = { is_society = hwarang }
		NOT = {
			has_global_flag = hwarang_disbanded
		}
	}

	immediate = {
		if = {
			limit = {
				is_landed = yes
			}
			
			# Pick someone under you to lead the warrior lodge...
			
			any_courtier_or_vassal = {
				limit = {
					ai = yes
					is_in_society = no
					has_secret_religion = no
					is_adult = yes
					block_general_event_trigger = no
					
					NOT = {
						has_character_modifier = not_interested_in_hwarang
					}
					has_hwarang_prerequisites = yes
				}
				
				score_value = { #Prioritizes selecting some kind of warrior...
					value = 10
					warrior_recruit_score = yes # TIANXIATODO: Improve?
					warrior_lodge_revival_selection_score = yes # TIANXIATODO: Improve?
				}
				
				join_society = hwarang
				
				if = {
					limit = {
						NOT = {
							society = {
								leader = { always = yes }
							}
						}
					}

						warrior_lodge_revival_selection_score = yes # TIANXIATODO: Improve?
					}
					
					join_society = hwarang
					
					if = {
						limit = {
							NOT = {

									leader = { always = yes }
								}
							}
						}

# AI rank up
character_event = {
	id = soh.808
	
	hide_window = yes
	is_triggered_only = yes # on_society_bi_yearly_pulse
	
	ai = yes
	min_age = 16
	
	#only_capable = yes
	prisoner = no

	trigger = {
		society_member_of = hwarang
		is_inaccessible_or_incapable_trigger = no
		society_can_rank_up = yes
	}

	immediate = {
		random_list = {
			100 = {}

			50 = {
				trigger = {
					society_rank == 1
				}
				society_rank_up = 1
				detract_society_currency_medium_effect = yes # at a worst, they are put at 0 currency
			}
			
			50 = {
				trigger = {
					society_rank == 2
				}
				society_rank_up = 1
				detract_society_currency_medium_effect = yes # at a worst, they are put at 0 currency
			}
		}
	}
}
