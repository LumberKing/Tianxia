# IDs 9901-10000 reserved### Japan

# Province flips from Yamato flips Japanese
province_event = {
	id = soh.9901
	desc = EVTDESC_SOH_9901
	picture = GFX_evt_castle_garden_japonic
	border = GFX_event_normal_frame_diplomacy

	trigger = {
		culture = yamato
		region = world_japan
		should_evolve_cultures_trigger = yes
		OR = {
			year >= 850
			has_global_flag = established_heian_kyo
		}
		year < 1100

		owner = {
			top_liege = {
				trigger_if = {
					limit = { culture = yamato }

					lower_real_tier_than = EMPEROR
					num_of_king_titles < 2
				}
				trigger_else = {
					culture = japanese
				}
			}

			religion_head = {
				NOT = { culture = yamato }
			}
		}
	}

	mean_time_to_happen = {
		months = 1200

		mult_modifier = {
			factor = 0.3

			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}

				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_SOH_9901

		trigger = {
			owner = {
				NOT = { culture = japanese }
			}
		}

		culture = japanese
	}

	option = {
		name = EVTOPTB_SOH_9901

		trigger = {
			owner = {
				culture = japanese
			}
		}

		culture = japanese
	}
}
