##### Regency factions

### Regency faction ultimatums and fallout

# Liege gets ultimatum
letter_event = {
	id = soh.6901
	desc = EVTDESC_SOH_6901
	
	is_triggered_only = yes
	
	trigger = {
		FROM = { in_faction = faction_regent }
	}
	
	immediate = {
		FROM = { 
			diplomatic_immunity = yes 
			any_faction_backer = {
				faction = faction_regent
				diplomatic_immunity = yes
			}
		}
	}
	
	option = {
		name = EVTOPTA_SOH_6901 # I suppose I have to accept this...
		
		regency_setup_nice_effect = yes
		
		# GRACETODO
		
		add_trait = under_regency
		
		set_global_flag = japanese_regency
		
		FROM = {
			add_trait = japanese_regent
			
			set_character_flag = ai_flag_never_transfer # No putting the PR under someone else!
			
			liege = {
				regent = {
					if = {
						limit = {
							NOT = {
								character = FROM
							}
						}
						remove_title = title_regent
					}
				}
				
				any_courtier_or_vassal = {
					limit = {
						has_minor_title = title_appointed_regent
						NOT = {
							character = FROM
						}
					}
					remove_title = title_appointed_regent
				}
			}
			
			give_minor_title = title_regent
			
			give_minor_title = title_appointed_regent
		}
		
		hidden_tooltip = {
			FROM = {
				any_faction_backer = {
					faction = faction_regent
					letter_event = { id = soh.6902 }
					diplomatic_immunity = no
				}
				
				letter_event = { id = soh.6903 }
				
				diplomatic_immunity = no
			}
		}
		
		ai_chance = {
			factor = 5
			
			trigger = {
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 1.0
					}
				}
			}
			
			mult_modifier = {
				factor = 1.25
				is_weak_trigger = yes
			}
			
			mult_modifier = {
				factor = 1.25
				trait = craven
			}
			
			mult_modifier = {
				factor = 1.25
				trait = humble
			}
			
			mult_modifier = {
				factor = 1.25
				trait = kind
			}
			
			mult_modifier = {
				factor = 1.25
				trait = charitable
			}
			
			mult_modifier = {
				factor = 1.25
				trait = patient
			}
			
			mult_modifier = {
				factor = 1.25
				trait = content
			}
		}
	}
	
	option = {
		name = EVTOPTB_SOH_6901 # Never!
		
		# GRACETODO

		if = {
			limit = {
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 1.0
					}
				}
			}

			random = {
				chance = 25

				character_event = {
					id = 45002 # Loyalists flock to the liege's banner
					months = 4
				}
			}
		}

		FROM = {
			show_scope_change = no

			letter_event = {
				id = soh.6904
				days = 3
				tooltip = EVTTOOLTIP8100
			}

			tooltip = {
				any_faction_backer = {
					faction = faction_regent
					count = 10
				}
			}

			hidden_effect = {
				if = {
					limit = {
						faction_power = {
							faction = faction_regent
							power < 0.75
						}
					}

					random_list = {
						50 = { }

						30 = {
							character_event = {
								id = 45000 # A great number of rebels flock to the revolter's banner
								days = 17
							}
						}

						20 = {
							character_event = {
								id = 45004 # Rebels flock to the revolter's banner
								days = 17
							}
						}
					}
				}

				else_if = {
					limit = {
						faction_power = {
							faction = faction_regent
							power < 1.5
						}
					}

					random_list = {
						75 = { }

						5 = {
							character_event = {
								id = 45000 # A great number of rebels flock to the revolter's banner
								days = 17
							}
						}

						20 = {
							character_event = {
								id = 45004 # Rebels flock to the revolter's banner
								days = 17
							}
						}
					}
				}
			}
		}
		
		ai_chance = {
			factor = 95
			
			mult_modifier = {
				factor = 1.25
				is_strong_trigger = yes
			}
			
			mult_modifier = {
				factor = 1.25
				trait = brave
			}
			
			mult_modifier = {
				factor = 1.25
				trait = proud
			}
			
			mult_modifier = {
				factor = 1.25
				trait = envious
			}
			
			mult_modifier = {
				factor = 1.25
				trait = greedy
			}
			
			mult_modifier = {
				factor = 1.25
				trait = wroth
			}
			
			mult_modifier = {
				factor = 1.25
				trait = ambitious
			}
			
			mult_modifier = {
				factor = 0.75
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 1.5
					}
				}
			}
			
			mult_modifier = {
				factor = 0.5
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 1.0
					}
				}
			}
			
			mult_modifier = {
				factor = 0.5
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 2.0
					}
				}
			}
			
			mult_modifier = {
				factor = 0.5
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 3.0
					}
				}
			}
			
			mult_modifier = {
				factor = 0.5
				FROM = {
					faction_power = {
						faction = faction_regent
						power >= 4.0
					}
				}
			}
		}
	}
}

# The regent is informed that the liege folded
letter_event = {
	id = soh.6903
	desc = EVTDESC_SOH_6903
	
	show_from_from = yes
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_SOH_6903 # Assuming indirect control!
		
		if = {
			limit = {
				NOT = {
					any_owned_bloodline = {
						has_bloodline_flag = regency_bloodline
						bloodline_is_active_for = ROOT
					}
				}
			}
			
			if = {
				limit = {
					is_female = no
				}
				create_bloodline = {
					type = regency_bloodline
				}
			}
			
			else = {
				create_bloodline = {
					type = regency_bloodline
					inheritance = matrilineal
				}
			}
		}
	}
}

# The prospective regent is informed that the liege isn't folding
letter_event = {
	id = soh.6904
	desc = EVTDESC_SOH_6904
	
	is_triggered_only = yes
	
	immediate = {
		any_faction_backer = {
			faction = faction_regent
			
			reverse_opinion = {
				modifier = opinion_fought_for_regent
				who = ROOT
				years = 20
			}
		}
	}
	
	option = {
		name = EVTOPTA_SOH_6904 # Then war it is!
		
		faction_regent = { # The target (title) of the faction
			reverse_war = {
				target = ROOT
				casus_belli = regency_faction_war
				faction = faction_regent
			}
		}
		
		clr_character_flag = faction_regent_ultimatum_taken
	}
}
