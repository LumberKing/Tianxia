character_event = {
	id = tianxia.1201
	desc = EVTDESC_TIANXIA_1201
	picture = GFX_evt_peasants

	is_triggered_only = yes # on_five_year_pulse -> random_events

	prisoner = no
	min_age = 16
	
	trigger = {
		is_inaccessible_or_incapable_trigger = no
		OR = {
			religion_openly_viet_or_reformed_trigger = yes
			AND = {
				culture_group = vietic
				has_chinese_eastern_religion_trigger = yes
			}
		}
		is_feudal = yes
		OR = {
			independent = yes
			higher_real_tier_than = duke
			controls_religion = yes
		}
		OR = {
			war = yes
			has_education_martial_trigger = yes
			has_lifestyle_martial_trigger = yes
			martial >= 10
		}
		is_devil_worshiper_trigger = no
		NOR = {
			has_global_flag = thuan_thien_blade_in_progress
			has_global_flag = blade_of_thuan_thien_exists
			has_global_flag = thuan_thien_exists
			has_character_flag = refused_thuan_thien
		}
	}
	
	immediate = {
		random_demesne_province = {
			preferred_limit = {
				OR = {
					port = yes
					borders_lake = yes
					borders_major_river = yes
				}
			}
			
			save_event_target_as = thuan_thien_province
		}
	}
	
	option = {
		name = EVTOPTA_TIANXIA_1201 # I accept your invitation!
		
		set_global_flag = thuan_thien_blade_in_progress
		
		hidden_tooltip = {
			random_list = {
				10 = {
					character_event = { id = tianxia.1202 days = 7 random = 7 } # Murder attempt - poison
				}
				10 = {
					character_event = { id = tianxia.1207 days = 7 random = 7 } # Murder attempt - dagger
				}
				20 = {
					character_event = { id = tianxia.1212 days = 7 random = 7 } # Fisherman serves man
				}
				20 = {
					character_event = { id = tianxia.1218 days = 7 random = 7 } # Board games
				}
				20 = {
					character_event = { id = tianxia.1221 days = 7 random = 7 } # An uneventful evening
				}
				10 = {
					trigger = {
						has_game_rule = {
							name = supernatural_events
							value = off
						}
					}
					character_event = { id = tianxia.1222 days = 7 random = 7 } # The Blade of Thuan Thien
				}
				10 = {
					trigger = {
						NOT = {
							has_game_rule = {
								name = supernatural_events
								value = off
							}
						}
					}
					character_event = { id = tianxia.1223 days = 7 random = 7 } # The Blade of Thuan Thien
				}
			}
		
			event_target:thuan_thien_province = {
				province_event = { id = tianxia.1232 days = 30 } # Flag cleanup
			}
		}
		
		ai_chance = {
			trigger = {
				NOT = {
					trait = proud
				}
			}
			
			factor = 100
		}
	}
	
	option = {
		name = EVTOPTB_TIANXIA_1201 # No thanks!
		
		add_trait_partially_proud_effect = yes
		
		set_character_flag = refused_thuan_thien
		
		ai_chance = {
			trigger = {
				NOT = {
					trait = humble
				}
			}
			
			factor = 100
		}
