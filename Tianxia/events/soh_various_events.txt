		}
	}

		name = EVTOPTB_SoH_180
		trigger = {
			NOT = {
				trait = paranoid

			}
		}
		if = {
			limit = { trait = paranoid }
			random = {
				chance = 50
				add_trait = wroth
			}
		}
		if = {
			limit = { NOT = { trait = paranoid } }
			random = {
				chance = 50
				add_trait = paranoid
			}
		}
	}
}

# Province bounce event
province_event = {
	id = SoH.3521

	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				any_neighbor_province = {
					religion_group = chinese_group
					NOT = { religion = ROOT }
				}
			}
			random_neighbor_province = {
				limit = {
					religion_group = chinese_group
					NOT = { religion = ROOT }
				}
				ROOT = {
					religion = PREV
				}
			}
		}
		if = {
			limit = {
				religion = confucian
				NOT = {
					any_neighbor_province = {
						religion_group = chinese_group
						NOT = { religion = ROOT }
					}
				}
			}
			random_list = {
				10 = { religion = amidist }
				10 = { religion = taoist }
			}
		}
		if = {
			limit = {
				religion = amidist
				NOT = {
					any_neighbor_province = {
						religion_group = chinese_group
						NOT = { religion = ROOT }
					}
				}
			}
			random_list = {
				10 = { religion = confucian }
				10 = { religion = taoist }
			}
		}
		if = {
			limit = {
				religion = taoist
				NOT = {
					any_neighbor_province = {
						religion_group = chinese_group
						NOT = { religion = ROOT }
					}
				}
			}
			random_list = {
				10 = { religion = confucian }
				10 = { religion = taoist }
