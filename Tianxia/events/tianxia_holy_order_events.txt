# Holy order stuff, largely copied from relevant vanilla holy orders, thus heavily based on work by Henrik FÃ¥hraeus (pagans) and Emil Tisander (non-pagans)
# The Shaolin Monks are around from the start; they're historical!
# The Sohei can be founded without any DLCs; they're a historical order!
# The Sons of Heaven and the Bushido Brotherhood require no DLC, as in the case of vanilla's Dharmic holy orders
# All new pagan holy orders require SoA as that's the most used vanilla check for pagan holy orders
# ONLY the pagan orders have donations, castle construction, expelling, and random character joining, as vanilla does not have that for the Dharmic holy orders
# The Inca holy order is found in sunrise_invasion.txt

# Ping the grandmaster
character_event = {
	id = tianxia_orders.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				e_japan = {
					has_holder = yes
					holder_scope = {
						OR = {
							religion = shinto
							religion = shugendo
							religion = koshin
							AND = {
								religion = buddhist
								culture_group = japonic
							}
						}
						OR = {
							AND = {
								completely_controls = c_yamato
								OR = {
									b_kofukuji = {
										has_holder = yes
										is_capital = no
									}
									b_todaiji = {
										has_holder = yes
										is_capital = no
									}
								}
							}
							AND = {
								completely_controls = c_omi
								b_enryakuji = {
									has_holder = yes
									is_capital = no
								}
							}
						}
					}
				}
			}
			e_japan = {
				holder_scope = {
					character_event = { id = tianxia_orders.4 } # See if Japan wants to vassalize them
				}
			}
		}
		else = {
			any_player = {
				limit = {
					OR = {
						religion = buddhist
						capital_scope = { region = world_japan }
						capital_scope = { region = world_japan_islands }
						capital_scope = { region = custom_yezo }
					}
				}
				narrative_event = { id = tianxia_orders.6 } # Inform people that the Sohei exist
			}
		}
	}
}

character_event = {
	id = tianxia_orders.118
	hide_window = yes
	
	min_age = 16
	only_playable = yes
	religion = dayak_pagan_reformed
	
	trigger = {
		d_headhunters = {
			holder_scope = {
				wealth = 300
				NOT = {
					realm_size = 10
				}
			}
		}
		
		any_demesne_province = {
			has_empty_holding = yes
		}
		
		has_dlc = "Sons of Abraham"
		
		NOT = { has_character_modifier = expelled_d_headhunters }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_headhunters
			}
		}
		NOT = { # China and pretenders shouldn't willingly let someone else build stuff in their land, and neither should the Tenno or a Shogun!
			top_liege = {
				OR = {
					government = chinese_imperial_government
					government = chinese_vassal_government
					government = eastern_imperial_government
					government = japanese_imperial_government
					government = japanese_bureaucracy_government
					government = japanese_feudal_government
					government = japanese_monastic_feudal_government
					government = divine_imperial_government
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			d_headhunters = {
				holder_scope = {
					NOT = { demesne_size = 1 }
				}
			}
		}
		modifier = {
			factor = 0.5
			d_headhunters = {
				holder_scope = {
					NOT = { demesne_size = 2 }
				}
			}
		}
		modifier = {
			factor = 2.0
			d_headhunters = {
				holder_scope = {
					num_of_count_titles = 1
				}
			}
		}
		modifier = {
			factor = 2.0
			d_headhunters = {
				holder_scope = {
					realm_size = 5
				}
			}
		}
	}
	
	immediate = {
		random_demesne_province = {
			limit = {
				has_empty_holding = yes
			}
			province_event = {
				id = tianxia_orders.119
			}
		}
	}
}
