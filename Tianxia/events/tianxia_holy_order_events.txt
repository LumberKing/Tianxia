##### Pagan holy orders

### The Defenders of the Kamuy

province_event = {
	id = tianxia.1001

	is_triggered_only = yes
	hide_window = yes

	immediate = {
		owner = {
			narrative_event = { id = tianxia.1002 }
		}
	}
}

# The birth of the Defenders of the Kamuy
narrative_event = {
	id = tianxia.1002
	title = EVTTITLE_TIANXIA_1002
	picture = GFX_evt_throne_room_norse_pagan
	portrait = event_target:target_grandmaster

	desc = {
		text = EVTDESC_TIANXIA_1002
		trigger = {
			NOT = {
				has_alternate_start_parameter = {
					key = religion_names
					value = random
				}
			}
		}
	}
	desc = {
		text = PAGAN_HOLY_ORDER_FORMED_ALT
		trigger = {
			has_alternate_start_parameter = {
				key = religion_names
				value = random
			}
		}
	}

	is_triggered_only = yes # on_reform_religion via event 670, base_on_action_events.txt
	major = yes
	hide_from = yes

	only_playable = yes
	has_dlc = "Sons of Abraham"

	trigger = {
		ainu_pagan = { has_flag = has_been_reformed }
		NOT = { is_title_active = d_defenders_of_the_kamuy }
	}

	immediate = {
		activate_title = {
			title = d_defenders_of_the_kamuy
			status = yes
		}

		d_defenders_of_the_kamuy = {
			religion = ainu_pagan_reformed

			save_persistent_event_target = {
				name = holy_order_religion
				scope = ainu_pagan_reformed
			}
		}

		create_character = {
			random_traits = no
			dynasty = random
			religion = ainu_pagan_reformed
			culture = ainu
			female = no
			age = 32
			add_trait = brilliant_strategist
			add_trait = zealous
			add_trait = humble
			add_trait = ambitious
			add_trait = strategist
			add_trait = holy_warrior
		}

		new_character = {
			if = {
				limit = {
					holy_order_accepts_male_commanders_trigger = yes
				}
				
				wealth = 500
				gain_title = d_defenders_of_the_kamuy
				set_government_type = order_government
				save_event_target_as = target_grandmaster
			}
			
			else = {
				create_character = {
					random_traits = no
					dynasty = random
					religion = ainu_pagan_reformed
					culture = ainu
					female = yes
					age = 32
					add_trait = brilliant_strategist
					add_trait = zealous
					add_trait = humble
					add_trait = ambitious
					add_trait = strategist
					add_trait = holy_warrior
				}
				
				new_character = {
					wealth = 500
					gain_title = d_defenders_of_the_kamuy
					set_government_type = order_government
					save_event_target_as = target_grandmaster
				}
			}
		}
		
		event_target:target_grandmaster = {
			while = {
				count = 7

				create_character = {
					random_traits = yes
					dynasty = random
					religion = ainu_pagan_reformed
					culture = ainu
					female = no
					age = 16
				}

				new_character = {
					random_list = {
						1 = { }
						1 = { add_age = 1 }
						1 = { add_age = 2 }
						1 = { add_age = 3 }
						1 = { add_age = 4 }
						1 = { add_age = 5 }
						1 = { add_age = 6 }
						1 = { add_age = 7 }
						1 = { add_age = 8 }
						1 = { add_age = 9 }
						1 = { add_age = 10 }
						1 = { add_age = 11 }
						1 = { add_age = 12 }
					}
				}
			}

			while = {
				count = 2

				create_character = {
					random_traits = yes
					dynasty = random
					religion = ainu_pagan_reformed
					culture = ainu
					female = yes
					age = 16
				}

				new_character = {
					random_list = {
						1 = { }
						1 = { add_age = 1 }
						1 = { add_age = 2 }
					}
				}
			}
		}
	}

	option = {
		name = {
			text = EVTOPTA_TIANXIA_1002
			trigger = { religion = ainu_pagan_reformed }
		}
		name = {
			text = EVTOPTB_SOA_2009
			trigger = {
				NOT = { religion = ainu_pagan_reformed }
			}
		}
	}
}

	hide_from = yes

		tai_pagan = { has_flag = has_been_reformed }
		NOT = { is_title_active = d_phi_sworn }

		activate_title = {
			title = d_phi_sworn
			status = yes
		}

			random_traits = no
			dynasty = random
			religion = tai_pagan_reformed
			culture = thai
			female = no
			age = 32
			add_trait = brilliant_strategist
			add_trait = zealous
			add_trait = humble
			add_trait = ambitious
			add_trait = strategist
			add_trait = holy_warrior
		}

		new_character = {
			if = {
				limit = {
					holy_order_accepts_male_commanders_trigger = yes
				}
				
				wealth = 500
				gain_title = d_phi_sworn
				set_government_type = order_government
				save_event_target_as = target_grandmaster

			else = {
				create_character = {
					random_traits = no

					set_government_type = order_government
					save_event_target_as = target_grandmaster

			while = {
				count = 7

					random_traits = yes
					dynasty = random
					religion = tai_pagan_reformed
					culture = thai
					female = no
					age = 16
				}

				new_character = {
					random_list = {
						1 = { }
						1 = { add_age = 1 }
						1 = { add_age = 2 }
						1 = { add_age = 3 }
						1 = { add_age = 4 }
						1 = { add_age = 5 }
						1 = { add_age = 6 }
						1 = { add_age = 7 }
						1 = { add_age = 8 }
						1 = { add_age = 9 }
						1 = { add_age = 10 }
						1 = { add_age = 11 }
						1 = { add_age = 12 }

					random_traits = yes
					dynasty = random
