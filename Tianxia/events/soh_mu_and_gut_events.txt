	is_triggered_only = yes

# Hidden event to determine which possible Court Mu you has been discovered
character_event = {
	id = soh.302
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		religion_openly_korean_or_reformed_trigger = yes # If we have converted, we no longer need a Court Mu
		
		NOT = {
			any_courtier_or_vassal = {
				has_minor_title = title_court_mu # If we've already got a Court Mu, we don't need another
			}
		}
	}
	
	immediate = {
		random_list = {
			10 = {
				character_event = { id = soh.303 } # Found possible Mu - healer
			}
			
			10 = {
				character_event = { id = soh.304 } # Found possible Mu - mystic
			}
			
			1 = {
				character_event = { id = soh.305 } # Found possible Mu - foreigner
			}
			
			10 = {
				character_event = { id = soh.306 } # Found possible Mu - charlatan
			}
			
			1 = {
				character_event = { id = soh.307 } # Found possible Mu - theologian
			}
			
			5 = {
				character_event = { id = soh.308 } # Found possible Mu - scholar
			}
			
			10 = {
				character_event = { id = soh.309 } # Found possible Mu - young commoner

				character_event = { id = soh.310 } # Found possible Mu - possessed
			}

				character_event = { id = soh.311 } # Found possible Mu - lunatic
			}
			
			10 = {
				character_event = { id = soh.312 } # Found possible Mu - zealot
			}
			
			5 = {
				character_event = { id = soh.313 } # Found possible Mu - drunkard
			}
			
			5 = {
				character_event = { id = soh.314 } # Found possible Mu - bad Learning character
			}
			
			10 = {
				character_event = { id = soh.315 } # Found possible Mu - okay Learning character
			}
			
			5 = {
				character_event = { id = soh.316 } # Found possible Mu - good Learning character
			}
			
			1 = {
				character_event = { id = soh.317 } # Found possible Mu - great Learning character
			}
			
			5 = {
				character_event = { id = soh.318 } # Found possible Mu - unremarkable character
			}
			
			1 = {
				trigger = {
					any_courtier = {
						is_landed = no
						dynasty = ROOT
						religion_openly_korean_or_reformed_trigger = yes
						is_married = no
						is_inaccessible_or_incapable_trigger = no
						age >= 12
						age <= 20
						
						OR = {
							AND = {
								religion = korean_pagan
								is_female = yes
							}
							
							AND = {
								is_female = no
								religion = korean_pagan_reformed
								korean_pagan_reformed = {
									has_religion_feature = religion_patriarchal
								}
							}
							
							AND = {
								religion = korean_pagan_reformed
								korean_pagan_reformed = {
									NOR = {
										has_religion_feature = religion_matriarchal
										has_religion_feature = religion_patriarchal
									}
								}
							}
						}
						
						OR = {
							trait = zealous
							trait = lunatic
							trait = possessed
							trait = ambitious
							trait = erudite
						}
					}
				}
				
				character_event = { id = soh.319 } # Found possible Mu - young relative
			}
			
			10 = {
				trigger = {
					any_vassal = {
						is_theocracy = yes
						is_inaccessible_or_incapable_trigger = no
						religion_openly_korean_or_reformed_trigger = yes
						age >= 12
						age <= 20
						
						OR = {
							AND = {
								religion = korean_pagan
								is_female = yes
							}
							
							AND = {
								is_female = no
								religion = korean_pagan_reformed
								korean_pagan_reformed = {
									has_religion_feature = religion_patriarchal
								}
							}
							
							AND = {
								religion = korean_pagan_reformed
								korean_pagan_reformed = {
									NOR = {
										has_religion_feature = religion_matriarchal
										has_religion_feature = religion_patriarchal
									}
								}
							}
						}
					}
				}
				character_event = { id = soh.320 } # Found possible Mu - temple vassal
			}
		}
	}

### Setup

# Court Mu informed of gut request
character_event = {
	id = soh.321
	desc = EVTDESC_SOH_321
	picture = GFX_evt_shaman
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	immediate = {
		FROM = {
			clr_character_flag = gut_extra_preparations
			clr_character_flag = gut_is_dodang_gut
			clr_character_flag = gut_is_ssitgim_gut
		}
		
		clr_character_flag = gut_extra_preparations
		clr_character_flag = gut_is_dodang_gut
		clr_character_flag = gut_is_ssitgim_gut
	}
	
	option = {
		name = EVTOPTA_SOH_321 # I'll begin the preparations at once!
		
		FROM = {
			character_event = { id = soh.328 days = 5 random = 10 } # Ping liege
		}
	}
	
	option = {
		trigger = {
			OR = {
				has_character_flag = court_mu_charlatan
				trait = greedy
				trait = zealous
				trait = drunkard
			}
		}
