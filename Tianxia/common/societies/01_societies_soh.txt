			NOT = { has_character_flag = society_join_block }
		}

			limit = { prisoner = yes }

	}
	
	show_society = {
		NOT = {
			has_game_rule = {
				name = devil_worshipers
				value = none
			}
		}
		age = 16
		OR = {
			religion = shinto
			religion = ryukyuan_pagan
			religion = ryukyuan_pagan_reformed
			religion = ainu_pagan
			religion = ainu_pagan_reformed
			AND = {
				culture_group = japonic
				religion = buddhist
			}
		}
	}
	
	potential = { 		
		OR = {
			religion = shinto
			religion = ryukyuan_pagan
			religion = ryukyuan_pagan_reformed
			religion = ainu_pagan
			religion = ainu_pagan_reformed
			AND = {
				culture_group = japonic

			}
		}
	}
	
	society_rank = {

	society_rank = { 
		level = 2
		limit = 50

			is_visible = {
				OR = {
					society_member_of = the_ghost_gatherers
