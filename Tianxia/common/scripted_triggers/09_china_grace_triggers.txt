china_forbids_war_trigger = { # Check for whether China could block a war due to a peace treaty, tributary relationship, etc.
	# Defender is China
	trigger_if = {
		limit = {
			FROM = {
				is_chinese_emperor_trigger = yes
			}
		}
		
		ROOT = {
			OR = {
				has_character_modifier = peace_deal_with_china
				any_liege = {
					has_character_modifier = peace_deal_with_china
				}
			}
		}
	}
	
	# Defender is a tributary of China
	trigger_else_if = {
		limit = {
			FROM = {
				OR = {
					is_tributary = {
						type = chinese_imperial_tributary
					}
					is_tributary = {
						type = chinese_protectorate
					}
					is_tributary = {
						type = chinese_client_state
					}
				}
				
				suzerain = {
					is_chinese_emperor_trigger = yes
				}
			}
		}
		
		ROOT = {
			OR = {
				has_character_modifier = peace_deal_with_china
				any_liege = {
					has_character_modifier = peace_deal_with_china
				}
			}
		}
	}
	
	# Attacker is China
	trigger_else_if = {
		limit = {
			ROOT = {
				is_chinese_emperor_trigger = yes
			}
		}
		
		FROM = {
			has_character_modifier = peace_deal_with_china
		}
	}
	
	# Attacker is a vassal of China
	trigger_else_if = {
		limit = {
			ROOT = {
				top_liege = {
					is_chinese_emperor_trigger = yes
				}
			}
		}
		
		FROM = {
			OR = {
				has_character_modifier = peace_deal_with_china
				any_liege = {
					has_character_modifier = peace_deal_with_china
				}
				
				AND = {
					OR = {
						is_tributary = {
