is_born_under_Aries_trigger = {
	trigger_if = {
		limit = { month_of_birth == 3 }
		day_of_birth >= 20
	}
	trigger_else = {
		month_of_birth == 4
		day_of_birth < 19
	}
}

	# Checks whether the ingredients selected are same type, such as two Hibiscus (if so, returns true)...

		has_artifact_flag = selected_ingredient_1

		ROOT = {
			any_artifact = {

			}
		}
	}
