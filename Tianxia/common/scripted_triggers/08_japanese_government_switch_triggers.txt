can_change_to_japanese_feudal_trigger = {
	trigger_if = {
		limit = {
			independent = yes
		}
		
		prestige >= 500
	}
	
	trigger_else_if = {
		limit = {
			top_liege = {
				government = japanese_feudal_government
			}
		}
		
		OR = {
			top_liege = {
				dynasty = ROOT
			}
			
			prestige >= 500
		}
	}
	
	trigger_else_if = {
		limit = {
			top_liege = {
				has_japanese_government_trigger = yes
			}
		}
		
		trigger_if = {
			limit = {
				any_owned_bloodline = {
					has_bloodline_flag = bloodline_japanese_feudal
					bloodline_is_active_for = ROOT
				}
			}
			
			custom_tooltip = {
				text = has_japanese_feudal_bloodline
				
				any_owned_bloodline = {
					has_bloodline_flag = bloodline_japanese_feudal
					bloodline_is_active_for = ROOT
				}
			}
			
			prestige >= 500
		}
		
		trigger_else_if = {
			limit = {
				top_liege = {
					is_under_regency_trigger = yes
				}
			}
			
			prestige >= 1000
		}
		
		trigger_else = {
			prestige >= 2500
		}
	}
	
	trigger_else_if = {
		limit = {
			top_liege = {
				has_chinese_government_trigger = yes
			}
		}
		
		trigger_if = {
			limit = {
				any_owned_bloodline = {
					has_bloodline_flag = bloodline_japanese_feudal
					bloodline_is_active_for = ROOT
				}
			}
			
			custom_tooltip = {
				text = has_japanese_feudal_bloodline
				
				any_owned_bloodline = {
					has_bloodline_flag = bloodline_japanese_feudal
					bloodline_is_active_for = ROOT
				}
			}
			
			prestige >= 1000
		}
		
		trigger_else_if = {
			limit = {
				top_liege = {
					OR = {
						has_character_modifier = mandate_of_heaven_2
						has_character_modifier = mandate_of_heaven_1
					}
				}
			}
			
			top_liege = {
				OR = {
					has_character_modifier = mandate_of_heaven_2
					has_character_modifier = mandate_of_heaven_1
				}
			}
			
			prestige >= 1000
		}
		
		trigger_else = {
			prestige >= 2500
		}
	}
	
	trigger_else = {
		prestige >= 1000
	}
}
