JD_artifact_discovery_possible_trigger = {
	OR = {
		is_close_to_china_or_along_the_silk_road_trigger = yes
		any_realm_province = {
			OR = {
				region = world_china
				region = world_korea
				region = world_vietnam
				region = world_balhae
			}
		}
		top_liege = {
			has_chinese_government_trigger = yes
		}
	}
}
