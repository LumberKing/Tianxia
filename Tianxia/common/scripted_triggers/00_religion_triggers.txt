could_be_temporal_religious_head_trigger = {
	trigger_if = {
		limit = { controls_religion = yes }

		OR = {
			religion_group = muslim

			trigger_if = {
				limit = { has_religion_features = yes }
				OR = {
					has_religion_feature = religion_temporal_head
					has_religion_feature = religion_divine_ancestry
				}
			}
			trigger_else = {
				religion = norse_pagan_reformed
			}
			
			religion = shinto
			religion = confucian # If they get a rel head down the line
			religion = taoist # If they get a rel head down the line
		}
	}
}

		limit = { PREV = { true_religion_group = muslim } }
