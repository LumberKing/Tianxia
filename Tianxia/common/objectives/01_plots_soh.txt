# Japanese regency stuff
plot_replace_permanent_regent = {
	type = realm_characters
	intrigue_plot = yes
	
	potential = {
		is_playable = yes
		can_have_japanese_bureaucracy_and_feudal_trigger = yes
		liege = {
			religion = shinto
			controls_religion = yes
			trait = under_regency
			NOT = {
				dynasty = PREV
			}
		}
		NOT = { trait = incapable }
		has_regent = no
	}
	
	target_potential = {
		trait = japanese_regent
		has_regent = yes
	}
	
	allow = {
		NOT = {
			has_non_aggression_pact_with = FROM # If we have an NAP, we should break that first!
		}
		OR = { # Either the current regent doesn't have the bloodline, or we are part of the bloodline
			NOT = {
				any_owned_bloodline= {
					has_bloodline_flag = regency_bloodline
					bloodline_is_active_for = PREV
				}
			}
			any_owned_bloodline= {
				has_bloodline_flag = regency_bloodline
				bloodline_is_active_for = PREV
				any_bloodline_member = {
					character = FROM
				}
			}
		}
	}
	
	allow_join = {
		liege = {
			OR = {
				trait = under_regency
				trait = japanese_regent
			}
		}
		# AI vassals won't join the plot unless they like the presumptive regent quite a bit more or if they are envious
		OR = {
			FROM = {
				ai = no
			}
			opinion_diff = {
				first = FROM
				second = FROMFROM
				value = 25
			}
			trait = envious
		}
		# Dynasty members of the regent will only overthrow the regent if they also share the presumptive regent's dynasty
		OR = {
			FROM = {
				ai = no
			}
			dynasty = FROMFROM
			NOT = {
				dynasty = ROOT
			}
		}
		# NAPs block joining, and being a friend or lover of the target while not being one of the presumptive regent also blocks it
		NOR = {
			has_non_aggression_pact_with = FROM # If we have an NAP, we should break that first!
			character = FROM # Don't plot against outselves!
			AND = {
				OR = {
					any_friend = {
						character = FROM
					}
					any_lover = {
						character = FROM
					}
				}
				FROM = {
					NOR = {
						any_friend = {
							character = FROMFROM
						}
						any_lover = {
							character = FROMFROM
						}
					}
				}
			}
		}
	}
	
	success = {
		FROM = {
			trait = japanese_regent
		}
	}
	
	abort = {
		OR = {
			FROM = {
				NAND = {
					is_playable = yes
					can_have_japanese_bureaucracy_and_feudal_trigger = yes
					liege = {
						religion = shinto
						controls_religion = yes
						trait = under_regency
						NOT = {
							dynasty = PREV
						}
					}
					NOT = { trait = incapable }
					has_regent = no
				}
			}
			NAND = {
				is_playable = yes
				trait = japanese_regent
				has_regent = yes
				NOT = {
					has_non_aggression_pact_with = FROM # If we have an NAP, we should break that first!
				}
				OR = { # Either the current regent doesn't have the bloodline, or we are part of the bloodline
					NOT = {
						any_owned_bloodline= {
							has_bloodline_flag = regency_bloodline
							bloodline_is_active_for = PREV
						}
					}
					any_owned_bloodline= {
						has_bloodline_flag = regency_bloodline
						bloodline_is_active_for = PREV
						any_bloodline_member = {
							character = FROM
						}
					}
				}
			}
		}
	}
	
	effect = {
		FROM = {
			if = {
				limit = {
					NOT = { diplomacy = 10 }
				}
				change_diplomacy = 1
			}
			any_plot_backer = {
				reverse_opinion = {
					modifier = opinion_grateful
					who = FROM
					years = 5
				}
			}
			# Clear flags
			hidden_tooltip = {
				clr_character_flag = plot_replace_permanent_regent_taken
			}
		}
	}
	
	chance = {
		factor = 50
		modifier = {
			factor = 0
			FROM = {
				OR = {
					trait = kind
					trait = content
					trait = humble
				}
				NOR = {
					trait = ambitious
					trait = proud
					trait = envious
				}
			}
		}
		modifier = {
			factor = 0
			FROM = {
				ai = yes
				war = yes
			}
		}
		modifier = {
			factor = 0.01
			FROM = {
				trait = content
			}
		}
		modifier = {
			factor = 0.2
			FROM = {
				OR = {
					trait = dull
					trait = slow
				}
			}
		}
		modifier = {
			factor = 0.2
			FROM = {
				trait = kind
			}
		}
		modifier = {
			factor = 0
			FROM = {
				ROOT = {
					reverse_opinion = {
						who = FROM
						value = 50
					}
				}
				OR = {
					trait = envious
					trait = deceitful
					trait = ambitious
				}
			}
		}		
		modifier = {
			factor = 0
			FROM = {
				ROOT = {
					reverse_opinion = {
						who = FROM
						value = 25
					}
				}
				NOT = {
					OR = {
						trait = envious
						trait = deceitful
						trait = ambitious
					}
				}
			}
		}
		modifier = {
			factor = 0.2
			FROM = {
				ROOT = {
					reverse_opinion = {
						who = FROM
						value = 0
					}
				}
				NOT = {
					OR = {
						trait = envious
						trait = deceitful
						trait = ambitious
					}
				}
			}
		}
		modifier = {
			factor = 0.2
			FROM = {
				ROOT = {
					reverse_opinion = {
						who = FROM
						value = 25
					}
				}
				OR = {
					trait = envious
					trait = deceitful
					trait = ambitious
				}
			}
		}
		modifier = {
			factor = 1.5
			FROM = {
				NOT = {
					ROOT = {
						reverse_opinion = {
							who = FROM
							value = -50
						}
					}
				}
			}
		}
		modifier = {
			factor = 1.5
			FROM = {
				NOT = {
					ROOT = {
						reverse_opinion = {
							who = FROM
							value = -75
						}
					}
				}
			}
		}
		modifier = {
			factor = 1.5
			FROM = {
				is_smart_trigger = yes
			}
		}
		modifier = {
			factor = 2.0
			FROM = {
				trait = envious
			}
		}
		modifier = {
			factor = 2.0
			FROM = {
				trait = deceitful
			}
		}
		modifier = {
			factor = 4.0
			FROM = {
				trait = ambitious
			}
		}
		# If the regent is low tier, make us more likely to act
		modifier = {
			factor = 2.0
			NOT = {
				tier = king
			}
		}
		modifier = {
			factor = 2.0
			NOR = {
				tier = king
				tier = duke
			}
		}
		# If we're a king, make us more likely to act
		modifier = {
			factor = 2
			FROM = {
				tier = king
			}
		}
	}
}
