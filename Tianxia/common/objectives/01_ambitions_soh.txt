obj_become_grand_chancellor = {
	type = character
	potential = {
		has_dlc = "Conclave"
		has_dlc = "Jade Dragon"
		
		is_adult = yes
		is_inaccessible_or_incapable_trigger = no
		independent = no
		
		liege = {
			independent = yes
			
			higher_real_tier_than = duke
			
			has_chinese_government_trigger = yes
		}
	}
	
	allow = {
		is_playable = yes
		is_feudal = yes
		
		controls_religion = no
		mercenary = no
		holy_order = no
		
		NOT = {
			any_owned_bloodline = {
				has_bloodline_flag = grand_chancellor_bloodline
				bloodline_is_active_for = PREV
			}
		}
		
		higher_real_tier_than = count
		
		OR = {
			prestige >= 2000
			liege = {
				has_regent = yes
			}
		}
	}
	
	chance = {
		factor = 100
		
		trigger = {
			NOT = {
				trait = content
			}
		}
		
		mult_modifier = {
			factor = 10
			liege = {
				is_rival = ROOT
			}
		}
		
		mult_modifier = {
			factor = 10
			liege = {
				any_voter = {
					has_minor_title = title_grand_chancellor
					is_rival = ROOT
				}
			}
		}
		
		mult_modifier = {
			factor = 3.0
			trait = ambitious
		}
		
		mult_modifier = {
			factor = 0.5
			trait = shy
		}
		
		mult_modifier = {
			factor = 1.5
			trait = gregarious
		}
		
		mult_modifier = {
			factor = 1.5
			trait = proud
		}
		
		mult_modifier = {
			factor = 0.5
			trait = humble
		}
		
		mult_modifier = {
			factor = 1.25
			OR = {
				diplomacy >= 10
				stewardship >= 10
				intrigue >= 10
				martial >= 10
				learning >= 10
			}
		}
		
		mult_modifier = {
			factor = 1.5
			OR = {
				diplomacy >= 16
				stewardship >= 16
				intrigue >= 16
				martial >= 16
				learning >= 16
			}
		}
		
		mult_modifier = { 
			factor = 10
			among_most_powerful_vassals = 5 
		}
		
		mult_modifier = { 
			factor = 0.1
			liege = {
				dynasty = ROOT
			}
			NOR = {
				trait = ambitious
				liege = {
					is_rival = ROOT
				}
			}
		}
		
		mult_modifier = { 
			factor = 0.1
			liege = {
				dynasty = ROOT
				is_close_relative = ROOT
			}
			
			NOR = {
				trait = ambitious
				liege = {
					is_rival = ROOT
				}
			}
		}
	}
	
	abort = {
		OR = {
			is_feudal = no
			is_incapable = yes
			controls_religion = yes
			mercenary = yes
			holy_order = yes
			
			independent = yes
			
			liege = {
				OR = {
					independent = no
					has_chinese_government_trigger = no
				}
			}
		}
	}
	
	abort_effect = {
		if = {
			limit = {
				government = chinese_vassal_government
				independent = no
			}
			
			prestige = -1000
		}
	}

	effect = {
		prestige = 1000
	}
	
	success = {
		custom_tooltip = {
			text = obj_become_grand_chancellor_success_tooltip
			
			any_owned_bloodline = {
				has_bloodline_flag = grand_chancellor_bloodline
			}
		}
	}
}

				}

			}
