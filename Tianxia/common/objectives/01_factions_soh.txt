		OR = {
			# Is not affected by a Crown Law title
			NOT = {
				crownlaw_title = {
					always = yes
				}
			}
			
			# The Crown Law title holder is my liege
			crownlaw_title = {

			crownlaw_title = {

						has_law = crown_authority_2

		}
		NOT = { has_character_modifier = faction_regent_timer }
	}

	allow = {

			primary_title = {
