japanese_reconquest = {
	name = CB_NAME_JAPANESE_RECONQUEST
	war_name = WAR_NAME_JAPANESE_RECONQUEST
	sprite = 8
	truce_days = 3650
	
	is_permanent = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	allowed_to_target_tributaries = no
	
	can_use_gui = {
		ROOT = {
			OR = {
				is_the_tenno_trigger = yes
				is_ryukyuan_tenno_trigger = yes
				
				AND = {
					tier = emperor
					prestige >= 100
				}
				
				AND = {
					tier = king
					prestige >= 500
				}
				
				prestige >= 1000
			}
			
			china_forbids_war_trigger = no
			
			NOR = {
				trait = treasure_fleet_preparation
				trait = treasure_fleet_voyage
			}
			
			is_under_regency_trigger = no 
		}
		
		FROM = {
			NOR = {
				is_the_tenno_trigger = yes
				is_ryukyuan_tenno_trigger = yes
				
				any_realm_lord = {
					is_the_tenno_trigger = yes
					is_ryukyuan_tenno_trigger = yes
				}
			}
		}
	}

	can_use = {
		ROOT = {
			has_japanese_government_trigger = yes
		}
		
		FROM = {
			independent = yes
		}
	}

	can_use_title = {
		OR = { # In Japan proper, an integrated area, or Japan's de jure
			region = world_japan # Part of the Japan region
			
			AND = {
				region = world_ryukyu
				has_global_flag = ryukyu_integrated
			}
			AND = {
				region = world_hokkaido
				has_global_flag = ezo_integrated
			}
			
			de_jure_liege_or_above = e_japan
		}
		
		any_direct_de_jure_vassal_title = { # Title must be inside holy war range
			OR = {
				holder_scope = {
					OR = {
						character = ROOT
						is_liege_or_above = ROOT
					}
				}
				
				location = {
					any_neighbor_province = {
						OR = {
							AND = {
								has_owner = yes
								owner = {
									OR = {
										character = ROOT
										is_liege_or_above = ROOT
									}
								}
							}
							
							AND = {
								is_land = no
								any_neighbor_province = {
									OR = {
										AND = {
											has_owner = yes
											owner = {
												OR = {
													character = ROOT
													is_liege_or_above = ROOT
												}
											}
										}
										
										AND = {
											is_land = no
											any_neighbor_province = {
												owner = {
													OR = {
														character = ROOT
														is_liege_or_above = ROOT
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}
	
	on_add = {
		ROOT = {
			if = {
				limit = {
					is_the_tenno_trigger = no
					is_ryukyuan_tenno_trigger = no
				}
				
				if = {
					limit = {
						tier = emperor
					}
					prestige = -100
				}
				
				else = {
					if = {
						limit = {
							higher_tier_than = duke
						}
						prestige = -500
					}
					
					else = {
						prestige = -1000
					}
				}
			}
		}
		
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}
	
	is_valid = {
		ROOT = {
			has_japanese_government_trigger = yes
		}
		
		FROM = {
			independent = yes
			NOR = {
				is_the_tenno_trigger = yes
				is_ryukyuan_tenno_trigger = yes
				
				any_realm_lord = {
					is_the_tenno_trigger = yes
					is_ryukyuan_tenno_trigger = yes
				}
			}
		}
	}
	
	on_success = {
		attacker = {
			participation_scaled_prestige = 100
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = { 
				participation_scaled_prestige = 100
			}
		}
		
		defender = {
			prestige = -100
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_medium_effect = yes
			}
		}
	}

	on_success_title = {
		custom_tooltip = { 
			text = other_invasion_succ_tip
			hidden_tooltip = {
				attacker = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = defender
						type = invasion
					}
				}
			}
		}
		
		if = {
			limit = {
				holder_scope = {
					character = defender
				}
			}
			
			usurp_title_only = {
				target = attacker
				type = invasion
			}
		}
	}

	on_fail = {
		attacker = {
			prestige = -150
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_small_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_small_effect = yes
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			transfer_scaled_wealth = {
				to = defender
				value = 4.0
			}
			
			prestige = -200
		}
		
		defender = {
			participation_scaled_piety = 100
			participation_scaled_prestige = 200
			
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = { 
				participation_scaled_piety = 100
				participation_scaled_prestige = 200
				
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_medium_effect = yes
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = {
		factor = 1
		mult_modifier = {
			factor = 5
			ROOT = {
				OR = {
					is_the_tenno_trigger = yes
					is_ryukyuan_tenno_trigger = yes
				}
				
				NOT = {
					has_ambition = obj_see_the_realm_prosper
				}
			}
		}
		
		mult_modifier = {
			factor = 2
			FROM = {
				could_have_japanese_government_trigger = no
			}
			
			ROOT = {
				NOT = {
					has_ambition = obj_see_the_realm_prosper
				}
			}
		}
	}
}

japanese_subjugation = {
	name = CB_NAME_PAGAN_SUBJUGATION
	war_name = WAR_NAME_PAGAN_SUBJUGATION
	sprite = 8
	truce_days = 365

	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	allowed_to_target_tributaries = no
	
	sort_priority = 1000
	
	can_use_gui = {
		ROOT = {
			piety >= 500
			
			has_actual_regency_trigger = no
			
			china_forbids_war_trigger = no
			
			NOR = {
				trait = treasure_fleet_preparation
				trait = treasure_fleet_voyage
			}
		}
		
		FROM = {
			NOR = {
				is_the_tenno_trigger = yes
				is_ryukyuan_tenno_trigger = yes
				
				any_realm_lord = {
					OR = {
						is_the_tenno_trigger = yes
						is_ryukyuan_tenno_trigger = yes
					}
				}
			}
		}
	}
	
	on_add = {
		ROOT = {
			piety = -500
		}
		
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}
	
	can_use = {
		ROOT = {
			OR = {
				independent = yes
				real_tier = king
				i_am_appropriate_regent_trigger = yes
				is_the_tenno_trigger = yes
				is_ryukyuan_tenno_trigger = yes
			}
			
			has_japanese_government_trigger = yes
		}
		
		FROM = {
			independent = yes
			
			has_japanese_government_trigger = yes
			
			NOT = {
				any_war = {
					defender = { character = PREVPREV }
					using_any_subjugation_cb_trigger = yes
				}
			}
		}
	}
	
	can_use_title = {
		ROOT = {
			OR = {
				has_landed_title = PREV
				
				culture = FROM
				
				AND = {
					religion = FROM
					culture_group = FROM
				}
				
				AND = {
					is_the_tenno_trigger = yes
					FROM = {
						religion = shinto
					}
				}
				
				AND = {
					is_ryukyuan_tenno_trigger = yes
					FROM = {
						religion = ryukyuan_pagan_reformed
					}
				}
			}
		}
		
		# Only one subjugation per lifetime normally
		ROOT = {
			OR = {
				NOT = {
					has_character_modifier = launched_subjugation
				}
				
				AND = {
					has_ambition = obj_become_king
					capital_scope = {
						kingdom = {
							title = PREVPREVPREV
						}
					}
				}
				
				AND = {
					OR = {
						is_the_tenno_trigger = yes
						is_ryukyuan_tenno_trigger = yes
					}
					FROM = {
						religion = ROOT
					}
				}
			}
		}

		FROM = {
			any_realm_province = {
				de_jure_liege_or_above = PREVPREV
			}
		}
		
		# The attacker needs at least one county in the target kingdom, or a border, or be only two sea zones away from one of your counties
		any_direct_de_jure_vassal_title = {
			any_direct_de_jure_vassal_title = {
				OR = {
					holder_scope = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
					
					location = {
						any_neighbor_province = {
							OR = {
								AND = {
									has_owner = yes
									owner = {
										OR = {
											character = ROOT
											is_liege_or_above = ROOT
										}
									}
								}
								
								AND = {
									is_land = no
									any_neighbor_province = {
										OR = {
											AND = {
												has_owner = yes
												owner = {
													OR = {
														character = ROOT
														is_liege_or_above = ROOT
													}
												}
											}
											
											AND = {
												is_land = no
												any_neighbor_province = {
													owner = {
														OR = {
															character = ROOT
															is_liege_or_above = ROOT
														}
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}

		# Defender
		FROM = {
			num_of_realm_counties = {
				value >= 1
				title = PREV
			}
		}
	}
	
	is_valid = {
		ROOT = {
			has_japanese_government_trigger = yes
		}
		
		FROM = {
			has_japanese_government_trigger = yes
			
			NOR = {
				is_the_tenno_trigger = yes
				is_ryukyuan_tenno_trigger = yes
				
				any_realm_lord = {
					OR = {
						is_the_tenno_trigger = yes
						is_ryukyuan_tenno_trigger = yes
					}
				}
			}
		}
	}
	
	is_valid_title = {
		FROM = {
			any_realm_province = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}
	
	on_add_title = {
		attacker = {
			show_scope_change = no

			if = {
				limit = {
					trigger_if = {
						limit = { has_ambition = obj_become_king }

						capital_scope = {
							kingdom = {
								NOT = { title = PREVPREVPREV }
							}
						}
					}
				}

				add_character_modifier = {
					name = launched_subjugation
					years = 1000
					hidden = yes
				}
			}
		}
	}

	on_success = {
		attacker = {
			participation_scaled_prestige = 200
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 200
			}
		}
		
		defender = {
			prestige = -200
		}
	}
	
	on_success_title = {
		# Usurp the target kingdom itself, if the enemy holds it
		if = {
			limit = {
				holder = defender

				attacker = {
					lower_real_tier_than = EMPEROR
				}
			}

			attacker = {
				show_scope_change = no

				usurp_title = {
					target = PREV
					type = invasion
				}
			}
		}

		if = {
			# Take the enemy's duchies within the target kingdom, if we are the same or lower tier
			limit = {
				attacker = {
					NOT = { higher_real_tier_than = defender }
				}
			}

			defender = {
				show_scope_change = no

				any_demesne_title = {
					limit = {
						real_tier = DUKE

						OR = {
							title = PREVPREV
							de_jure_liege_or_above = PREVPREV
						}
					}

					attacker = {
						show_scope_change = no

						usurp_title = {
							target = PREV
							type = invasion
						}
					}
				}
			}
		}

		# Have I become the vassal of the loser due to taking a duchy or kingdom?
		if = {
			limit = {
				attacker = {
					is_vassal_or_below_of = defender
				}
			}

			attacker = {
				set_defacto_liege = THIS
			}
		}

		# Vassalize the rest
		custom_tooltip = {
			text = pagan_subjugation_tip

			attacker = {
				subjugate_or_take_under_title = { # If the target only has territory within the kingdom, he is simply vassalized
					title = PREV
					enemy = defender
				}
			}
		}
	}

	on_fail = {
		defender = {
			participation_scaled_prestige = 100
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 100
			}
		}
		
		attacker = {
			prestige = -100
		}
	}
	
	on_invalidation = {
		attacker = {
			remove_character_modifier = launched_subjugation
			piety = 500
		}
	}

	on_reverse_demand = {
		attacker = {
			transfer_scaled_wealth = {
				to = defender
				value = 2.0
			}
			
			prestige = -200
		}
		
		defender = {
			participation_scaled_prestige = 200
		}
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 200
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = { 
		factor = 1
		
		mult_modifier = {
			factor = 0.9 # Prefer using CB's without on_add costs
		}
	}
}

imperial_claim = {
	name = CB_NAME_IMPERIAL_CLAIM
	war_name = WAR_NAME_IMPERIAL_CLAIM
	sprite = 1
	truce_days = 3650
	hostile_against_others = yes
	can_ask_to_join_war = no
	is_permanent = yes
	
	defender_unoccupied_warscore = yes

	allowed_to_target_tributaries = no
	
	#press_claim = yes

	infamy_modifier = 0.85
	
	sort_priority = 1000
	
	can_use = {
		ROOT = {
			is_feudal = yes
			
			is_possible_tenno_trigger = yes
			religion = shinto
		}
		
		FROM = {
			is_the_tenno_trigger = yes
			dynasty = ROOT
		}
	}
	
	can_use_gui = {
		check_if_crusader_trigger = yes
		
		ROOT = {
			has_actual_regency_trigger = no
			piety >= 500
			
			china_forbids_war_trigger = no # Chinese Peace Deal or tributary relationship blocks CB
			
			NOR = {
				trait = treasure_fleet_preparation
				trait = treasure_fleet_voyage
			}
		}
		
		FROM = {
			OR = {
				liege = {
					any_vassal = {
						character = ROOT
					}
				}
				
				AND = {
					independent = yes
					ROOT = {
						independent = yes
					}
				}
			}
			
			NOT = {
				any_war = {
					OR = {
						using_cb = shogunate_war
						using_cb = imperial_claim_faction
						using_cb = imperial_claim
					}
				}
			}
		}
	}
	
	is_valid = {
		ROOT = {
			is_feudal = yes
			
			religion = shinto
			
			is_possible_tenno_trigger = yes
			
			NOT = {
				trait = under_regency2 # Nice try!
			}
		}
		
		FROM = {
			is_the_tenno_trigger = yes
			
			OR = {
				liege = {
					any_vassal = {
						character = ROOT
					}
				}
				
				AND = {
					independent = yes
					ROOT = {
						independent = yes
					}
				}
			}
		}
	}

	on_add = {
		attacker = {
			piety = -500
		}
		
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success = {
		attacker = {
			show_scope_change = no
			
			participation_scaled_prestige = 1000
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
		}
		
		defender = {
			show_scope_change = no
			
			prestige = -1000
			
			abdicate_to = attacker
		}
	}

	on_fail = {
		attacker = {
			show_scope_change = no
			prestige = -500
		}
		
		defender = {
			show_scope_change = no
			
			participation_scaled_prestige = 500
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 500
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			
			prestige = -1000
			
			if = {
				limit = {
					liege = {
						any_vassal = {
							character = defender
						}
					}
				}
				
				abdicate_to = defender
				
				imprison = defender
			}
		}
		
		defender = {
			show_scope_change = no
			
			participation_scaled_prestige = 1000
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = {
		factor = 1
		
		trigger = {
			ROOT = {
				NOR = {
					trait = content
					is_friend = FROM
					is_lover = FROM
				}
			}
		}
		
		mult_modifier = {
			factor = 10
			ROOT = {
				trait = ambitious
			}
		}
		
		mult_modifier = {
			factor = 0.1
			FROM = {
				trait = under_regency
			}
		}
		
		mult_modifier = {
			factor = 100
			ROOT = {
				is_rival = FROM
			}
		}
	}
}

	name = CB_NAME_PAGAN_SUBJUGATION
	war_name = WAR_NAME_PAGAN_SUBJUGATION
	sprite = 8
	truce_days = 365
	
	hostile_against_others = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	
	allowed_to_target_tributaries = no
	is_holy_war = yes # Co-religionists can join
	
	sort_priority = 1000
	
	can_use_gui = {
		ROOT = {
