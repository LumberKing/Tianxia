chinese_reconquest = {
	name = CB_NAME_CHINESE_RECONQUEST
	war_name = WAR_NAME_CHINESE_RECONQUEST
	sprite = 8
	truce_days = 3650
	
	is_permanent = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	allowed_to_target_tributaries = no
	
	can_use_gui = {
		ROOT = {
			OR = {
				is_chinese_emperor_trigger = yes
				
				AND = {
					government = chinese_imperial_government
					prestige >= 100
				}
				
				AND = {
					higher_real_tier_than = duke
					prestige >= 500
				}
				
				prestige >= 1000
			}
			
			china_forbids_war_trigger = no
			
			NOR = {
				trait = treasure_fleet_preparation
				trait = treasure_fleet_voyage
			}
		}
	}

	can_use = {
		ROOT = {
			is_feudal = yes
			NOT = {
				same_realm = FROM
			}
			
			OR = {
				government = chinese_imperial_government
				
				AND = {
					OR = {
						government = eastern_imperial_government
						government = chinese_vassal_government
					}
					
					culture_group = chinese_group
					
					independent = yes
					
					OR = {
						has_character_flag = mandate_of_heaven_uprising # A successful Mandate revolter can try to take over more of China
						
						e_china = {
							OR = {
								has_holder = no
								holder_scope = {
									is_chinese_emperor_trigger = no
								}
							}
						}
					}
				}
			}
			
			mercenary = no
			holy_order = no
		}
	}

	can_use_title = {
		OR = {
			de_jure_liege_or_above = e_china # Currently part of de jure China
			region = world_china
		}
		
		any_direct_de_jure_vassal_title = { # Title must be inside holy war range (to keep pretenders from launching wars from a great distance)
			OR = {
				holder_scope = {
					OR = {
						character = ROOT
						is_liege_or_above = ROOT
					}
				}
				
				location = {
					any_neighbor_province = {
						OR = {
							AND = {
								has_owner = yes
								owner = {
									OR = {
										character = ROOT
										is_liege_or_above = ROOT
									}
								}
							}
							
							AND = {
								is_land = no
								any_neighbor_province = {
									OR = {
										AND = {
											has_owner = yes
											owner = {
												OR = {
													character = ROOT
													is_liege_or_above = ROOT
												}
											}
										}
										
										AND = {
											is_land = no
											any_neighbor_province = {
												owner = {
													OR = {
														character = ROOT
														is_liege_or_above = ROOT
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}
	
	on_add = {
		ROOT = {
			if = {
				limit = {
					is_chinese_emperor_trigger = no
				}
				
				if = {
					limit = {
						government = chinese_imperial_government
					}
					prestige = -100
				}
				
				else = {
					if = {
						limit = {
							higher_tier_than = duke
						}
						prestige = -500
					}
					
					else = {
						prestige = -1000
					}
				}
			}
		}
		
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}
	
	is_valid = {
		ROOT = {
			independent = yes
			has_chinese_government_trigger = yes
			NOT = {
				same_realm = FROM
			}
		}
	}
	
	on_success = {
		attacker = {
			participation_scaled_prestige = 100
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = { 
				participation_scaled_prestige = 100
			}
		}
		
		defender = { prestige = -100 }
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_medium_effect = yes
			}
		}
	}

	on_success_title = {
		custom_tooltip = { 
			text = other_invasion_succ_tip
			hidden_tooltip = {
				attacker = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = defender
						type = invasion
					}
				}
			}
		}
		
		if = {
			limit = {
				holder_scope = {
					character = defender
				}
			}
			
			usurp_title_only = {
				target = attacker
				type = invasion
			}
		}
	}

	on_fail = {
		attacker = {
			prestige = -150
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_small_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_small_effect = yes
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			
			prestige = -200
		}
		
		defender = {
			participation_scaled_piety = 100
			participation_scaled_prestige = 200
			
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = { 
				participation_scaled_piety = 100
				participation_scaled_prestige = 200
				
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_medium_effect = yes
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = {
		factor = 1
		mult_modifier = {
			factor = 2
			ROOT = {
				government = chinese_imperial_government
				NOT = {
					has_ambition = obj_see_the_realm_prosper
				}
			}
		}
		
		mult_modifier = {
			factor = 2
			ROOT = {
				is_chinese_emperor_trigger = yes
				NOT = {
					has_ambition = obj_see_the_realm_prosper
				}
			}
		}
		
		mult_modifier = {
			factor = 2
			e_china = {
				OR = {
					has_holder = no
					holder_scope = {
						NOT = {
							is_chinese_emperor_trigger = yes
						}
					}
				}
			}
			
			ROOT = {
				NOT = {
					has_ambition = obj_see_the_realm_prosper
				}
			}
		}
		
		mult_modifier = {
			factor = 2
			ROOT = {
				NOT = {
					has_ambition = obj_see_the_realm_prosper
				}
			}
			
			FROM = {
				is_tribal = no
				is_nomadic = no
				has_chinese_government_trigger = no
			}
		}
	}
}

# Chinese-style government subjugation CB
chinese_subjugation = {
	name = CB_NAME_PAGAN_SUBJUGATION
	war_name = WAR_NAME_PAGAN_SUBJUGATION
	sprite = 8
	truce_days = 365

	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	allowed_to_target_tributaries = no
	
	sort_priority = 1000
	
	can_use_gui = {
		ROOT = {
			piety >= 500
			
			china_forbids_war_trigger = no
			
			NOR = {
				trait = treasure_fleet_preparation
				trait = treasure_fleet_voyage
			}
		}
	}
	
	on_add = {
		ROOT = {
			piety = -500
		}
		
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}
	
	can_use = {
		ROOT = {
			OR = {
				independent = yes
				tier = king
			}
			has_chinese_government_trigger = yes
			
			mercenary = no
			holy_order = no
		}
		
		FROM = {
			independent = yes
			has_chinese_government_trigger = yes
			
			NOT = {
				any_war = {
					defender = { character = PREVPREV }
					using_any_subjugation_cb_trigger = yes
				}
			}
		}
	}
	
	can_use_title = {
		OR = {
			ROOT = {
				OR = {
					has_landed_title = PREV
					culture = FROM
					
					AND = {
						religion = FROM
						culture_group = FROM
					}
					
					custom_tooltip = {
						text = ci_bloodline_founder
						any_owned_bloodline = {
							OR = {
								has_bloodline_flag = real_chinese_imperial_bloodline
								has_bloodline_flag = pretender_chinese_imperial_bloodline
							}
							founder = {
								character = ROOT
							}
						}
					}
					
					has_character_modifier = mandate_of_heaven_5
				}
			}
			
			FROM = {
				has_character_modifier = mandate_of_heaven_1
			}
			
			AND = { # Chinese Imperial empires can always use this against one another for Chinese land
				ROOT = {
					government = chinese_imperial_government
				}
				FROM = {
					government = chinese_imperial_government
				}
				
				OR = {
					any_direct_de_jure_vassal_title = {
						region = world_china
					}
					de_jure_liege_or_above = e_china
				}
			}
		}
		
		# Only one subjugation per lifetime normally
		ROOT = {
			OR = {
				NOT = {
					has_character_modifier = launched_subjugation
				}
				
				has_character_modifier = mandate_of_heaven_5
				
				custom_tooltip = {
					text = ci_bloodline_founder
					any_owned_bloodline = {
						OR = {
							has_bloodline_flag = real_chinese_imperial_bloodline
							has_bloodline_flag = pretender_chinese_imperial_bloodline
						}
						founder = {
							character = ROOT
						}
					}
				}
				
				AND = {
					has_ambition = obj_become_king
					capital_scope = {
						kingdom = {
							title = PREVPREVPREV
						}
					}
				}
			}
		}

		FROM = {
			any_realm_province = {
				de_jure_liege_or_above = PREVPREV
			}
		}
		
		# The attacker needs at least one county in the target kingdom, or a border, or be only two sea zones away from one of your counties
		any_direct_de_jure_vassal_title = {
			any_direct_de_jure_vassal_title = {
				OR = {
					holder_scope = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
					
					location = {
						any_neighbor_province = {
							OR = {
								AND = {
									has_owner = yes
									owner = {
										OR = {
											character = ROOT
											is_liege_or_above = ROOT
										}
									}
								}
								
								AND = {
									is_land = no
									any_neighbor_province = {
										OR = {
											AND = {
												has_owner = yes
												owner = {
													OR = {
														character = ROOT
														is_liege_or_above = ROOT
													}
												}
											}
											
											AND = {
												is_land = no
												any_neighbor_province = {
													owner = {
														OR = {
															character = ROOT
															is_liege_or_above = ROOT
														}
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}

		# Defender
		FROM = {
			num_of_realm_counties = {
				value >= 1
				title = PREV
			}
		}
	}
	
	is_valid = {
		ROOT = {
			has_chinese_government_trigger = yes
		}
		
		FROM = {
			has_chinese_government_trigger = yes
		}
	}
	
	is_valid_title = {
		FROM = {
			any_realm_province = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	on_add_title = {
		attacker = {
			show_scope_change = no

			if = {
				limit = {
					trigger_if = {
						limit = { has_ambition = obj_become_king }

						capital_scope = {
							kingdom = {
								NOT = { title = PREVPREVPREV }
							}
						}
					}
				}

				add_character_modifier = {
					name = launched_subjugation
					years = 1000
					hidden = yes
				}
			}
		}
	}

	on_success = {
		attacker = {
			participation_scaled_prestige = 200
			
			if = {
				limit = {
					this_has_super_cool_nickname_trigger = no
					age < 25
					defender = {
						higher_tier_than = duke
					}
				}
				
				give_nickname = nick_the_little_conqueror
			}
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
		
		defender = {
			prestige = -200
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
	}
	
	on_success_title = {
		# Usurp the target kingdom itself, if the enemy holds it
		if = {
			limit = {
				holder = defender

				attacker = {
					lower_real_tier_than = EMPEROR
				}
			}

			attacker = {
				show_scope_change = no

				usurp_title = {
					target = PREV
					type = invasion
				}
			}
		}

		if = {
			# Take the enemy's duchies within the target kingdom, if we are the same or lower tier
			limit = {
				attacker = {
					NOT = { higher_real_tier_than = defender }
				}
			}

			defender = {
				show_scope_change = no

				any_demesne_title = {
					limit = {
						real_tier = DUKE

						OR = {
							title = PREVPREV
							de_jure_liege_or_above = PREVPREV
						}
					}

					attacker = {
						show_scope_change = no

						usurp_title = {
							target = PREV
							type = invasion
						}
					}
				}
			}
		}
		
		# Have I become the vassal of the loser due to taking a duchy or kingdom?
		if = {
			limit = {
				attacker = {
					is_liege_or_above = defender
				}
			}
			
			attacker = {
				set_defacto_liege = THIS
			}
		}
		
		# Vassalize the rest
		custom_tooltip = {
			text = pagan_subjugation_tip
			hidden_tooltip = {
				attacker = {
					subjugate_or_take_under_title = { # If the target only has territory within the kingdom, he is simply vassalized
						title = PREV
						enemy = defender
					}
				}
			}
		}
	}

	on_fail = {
		defender = {
			participation_scaled_prestige = 100
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		
		attacker = {
			prestige = -100
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_large_effect = yes
			}
			roman_victory_triumph_effect = yes
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_large_effect = yes
			}
		}
	}
	
	on_invalidation = {
		attacker = {
			remove_character_modifier = launched_subjugation
			piety = 500
		}
	}

	on_reverse_demand = {
		attacker = {
			prestige = -200
			
			transfer_scaled_wealth = {
				to = defender
				value = 2.0
			}
		}
		
		defender = {
			participation_scaled_prestige = 200
			roman_victory_triumph_effect = yes
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 200
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = { 
		factor = 1
		
		mult_modifier = {
			factor = 0.9 # Prefer using CB's without on_add costs
		}
		
		# Founders of Chinese dynasties should try to expand more
		mult_modifier = {
			factor = 5
			ROOT = {
				any_owned_bloodline = {
					OR = {
						has_bloodline_flag = pretender_chinese_imperial_bloodline
						has_bloodline_flag = real_chinese_imperial_bloodline
					}
					
					founder = {
						character = ROOT
					}
				}
			}
		}
	}
}

# Tributary tier war
chinese_tributary_tier_war = { # Check
	name = CB_NAME_TRIBUTARY_TIER
	war_name = WAR_NAME_TRIBUTARY_TIER
	sprite = 25
	truce_days = 3650
	
	can_ask_to_join_war = no
	is_permanent = no
	
	defender_unoccupied_warscore = yes
	hostile_against_others = yes
	
	allow_whitepeace = no # You win, or you lose
	allowed_to_target_tributaries = yes # Otherwise, things wouldn't work
	
	can_use = {
		always = no # Triggered from events only
	}
	
	# No need to check Grace penalties or peace deals!
	
	is_valid = {
		ROOT = {
			is_chinese_emperor_trigger = yes
		}
	}
	
	on_success = {
		attacker = {
			if = {
				limit = {
					defender = {
						tier = emperor
						is_nomadic = no
					}
				}
				prestige = 1000
			}
			
			else_if = {
				limit = {
					defender = {
						OR = {
							tier = king
							AND = {
								tier = emperor
								is_nomadic = yes
							}
						}
					}
				}
				
				prestige = 500
			}
			
			else = {
				prestige = 250
			}
		}
		
		defender = {
			increase_tributary_tier_or_vassalize_end_of_war_effect = yes
			
			
			if = {
				limit = {
					is_the_tenno_trigger = no
				}
				
				abdicate_to_most_liked_by = attacker
			}
			
			else = {
				abdicate = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_medium_effect = yes
			}
		}
	}
	
	on_reverse_demand = {
		attacker = {
			if = {
				limit = {
					defender = {
						tier = emperor
						is_nomadic = no
					}
				}
				prestige = -1000
			}
			
			else_if = {
				limit = {
					defender = {
						OR = {
							tier = king
							AND = {
								tier = emperor
								is_nomadic = yes
							}
						}
					}
				}
				
				prestige = -500
			}
			
			else = {
				prestige = -250
			}
			
			remove_tributary = defender
			
			defender = {
				save_event_target_as = defiant_tributary
			}
			
			opinion = {
				who = defender
				modifier = opinion_china_breakaway_tributary
			}
			
			any_tributary = {
				narrative_event = { id = soh.6432 } # Other tributaries get the option to downgrade
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		attacker = {
