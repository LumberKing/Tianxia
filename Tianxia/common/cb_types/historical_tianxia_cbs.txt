unification_of_korea = { # For the 936 start
	name = CB_NAME_UNIFICATION_OF_KOREA
	war_name = WAR_NAME_UNIFICATION_OF_KOREA
	diplo_view_region = world_korea
	sprite = 12
	truce_days = 3650
	
	is_permanent = no

	allowed_to_target_tributaries = no

	can_use = {
		always = no # Should only be triggered in script
	}

	is_valid = {
		ROOT = {
			OR = {
				has_landed_title = k_goryeo
				has_landed_title = k_silla
				has_landed_title = k_baekje
				#has_landed_title = k_balhae
			}
		}
		
		FROM = {
			OR = {
				has_landed_title = k_goryeo
				has_landed_title = k_silla
				has_landed_title = k_baekje
				#has_landed_title = k_balhae
			}
		}
	}

	on_success = {
		attacker = {
			participation_scaled_prestige = 1000
			
			if = {
				limit = {
					has_character_flag = is_taejo
				}
				create_bloodline = {
					type = taejo_bloodline
				}
			}
			
			else = {
				create_bloodline = {
					type = korean_bloodline # Generic
				}
			}
		}
		
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
		}
		
		defender = {
			prestige = -1000
			# In the unlikely event that the defender or a vassal has land outside the Korean Peninsula, we need to handle it
			any_realm_lord = {
				limit = {
					higher_tier_than = baron
				}
				
				if = {
					limit = {
						NOT = {
							any_demesne_title = {
								capital_scope = {
									NOT = {
										region = world_korea
									}
								}
							}
						}
					}
					
					opinion = {
						modifier = opinion_subjugated 
						who = ROOT
					}
					
					set_defacto_liege = attacker
				}
				
				else = {
					any_demesne_title = {
						limit = {
							capital_scope = {
								region = world_korea
							}
						}
						
						if = {
							limit = {
								tier = king
							}
							# Clean up the "Hu" in "Hubaekje"
							set_name = ""
							adjective = ""
							unsafe_destroy_landed_title = THIS
						}
						
						else = {
							usurp_title_plus_barony_if_unlanded = attacker
						}
					}
				}
			}
			
			if = {
				limit = {
					NOT = {
						any_demesne_title = {
							capital_scope = {
								NOT = {
									region = world_korea
								}
							}
						}
					}
				}
				
				any_demesne_title = {
					limit = {
						OR = {
							tier = king
							tier = emperor
						}
					}
					
					unsafe_destroy_landed_title = THIS
				}
				
				opinion = {
					modifier = opinion_subjugated 
					who = attacker
				}
				
				set_defacto_liege = attacker
			}
			
			else = {
				any_demesne_title = {
					limit = {
						capital_scope = {
							region = world_korea
						}
					}
					
					if = {
						limit = {
							OR = {
								tier = king
								tier = emperor
							}
						}
						
						unsafe_destroy_landed_title = THIS
					}
					
					else = {
						usurp_title_plus_barony_if_unlanded = attacker
					}
				}
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_large_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_large_effect = yes
			}
		}
	}
	
	on_fail = {
		attacker = {
			prestige = -500
		}
		
		defender = {
			participation_scaled_prestige = 500
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 100
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_medium_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_medium_effect = yes
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			transfer_scaled_wealth = {
				to = defender
				value = 2.0
			}
			
			prestige = -1000
		}
		
		defender = {
			participation_scaled_prestige = 2000
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			hidden_tooltip = {
				participation_scaled_prestige = 2000
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_large_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_large_effect = yes
			}
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

# Jin-Tang in 936
jin_tang_cb = {
	name = CB_NAME_CHINESE_DYNASTIC_WAR
	war_name = WAR_NAME_CHINESE_DYNASTIC_WAR
	sprite = 9
	truce_days = 3650
	diplo_view_region = world_china
	is_permanent = no # Only started from script

	allowed_to_target_tributaries = no
	allow_whitepeace = no # You win, or you lose
	
	# No need to check Chinese Peace Deals

	can_use = {
		always = no # Should only be triggered in script
	}
	
	is_valid = {
		ROOT = {
			has_landed_title = e_great_wall
			government = chinese_imperial_government
		}
		FROM = {
			has_landed_title = e_black_turtle
		}
	}

	on_success = {
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
		
		attacker = {
			participation_scaled_prestige = 1000
		}
		
		# 1: Usurp all vassals
		defender = {
			any_vassal = {
				limit = {
					higher_tier_than = baron
				}
				
				set_defacto_liege = attacker
			}
		}
		
		# 2: Take over the new empire and handle the renaming (assuming it is relevant), then give the attacker the defender as a vassal
		attacker = {
			set_character_flag = special_chinse_imperial_usurpation
			
			e_black_turtle = {
				usurp_title = attacker
				set_preferred_capital = c_bianjing
				save_event_target_as = pretender_title
				make_primary_title = yes
				clear_all_chinese_dynasty_flags_and_update_global_flags_effect = yes
				set_title_flag = pretender_chinese_empire
			}
			
			clr_character_flag = special_chinse_imperial_usurpation
			
			e_great_wall = {
				show_scope_change = no
				copy_pretender_empire_name_pretender_effect = yes
				set_name = ""
				adjective = ""
				unsafe_destroy_landed_title = THIS
				clear_all_chinese_dynasty_flags_and_update_global_flags_effect = yes
			}
			
			set_government_type = chinese_imperial_government
		}
		
		defender = {
			set_defacto_liege = attacker
			
			imprison = attacker
			
			add_trait = former_eoc_trait
		}
		
		attacker = {
			if = {
				limit = {
					any_realm_lord = {
						has_landed_title = c_bianjing
					}
				}
				
				c_bianjing = {
					usurp_title = attacker
					
					attacker = {
						capital = c_bianjing
					}
				}
			}
			
			if = {
				limit = {
					any_realm_lord = {
						has_landed_title = c_luoyang
					}
				}
				
				c_luoyang = {
					usurp_title = attacker
				}
			}
			
			if = {
				limit = {
					any_realm_lord = {
						has_landed_title = c_changan
					}
				}
				
				c_changan = {
					usurp_title = attacker
				}
			}
		}
		
		# 3: If the defender has the Heirloom Seal of the Realm, transfer it to the attacker
		defender = {
			if = {
				limit = {
					any_artifact = {
						has_flag = heirloom_seal_of_the_realm
					}
				}
				
				any_artifact = {
					limit = {
						has_flag = heirloom_seal_of_the_realm
					}
					
					transfer_artifact = {
						from = defender
						to = attacker
					}
				}
			}
		}
		
		# 4: Prestige for allies + potential Sixteen Prefectures event
		any_attacker = {
			limit = { NOT = { character = attacker } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 1000
			}
			
			if = {
				limit = {
					independent = yes
					tier = emperor
					government = chinese_imperial_government
					
					dynasty = 1045082 # Liao Dynasty
					
					any_owned_bloodline = {
						has_bloodline_flag = liao_china
					}
				}
				
				save_event_target_as = liao_emperor
				
				attacker = {
					character_event = { id = soh.7201 } # Sixteen Prefectures event
				}
			}
		}
		
		# Prestige loss for the defender
		defender = {
			prestige = -1000
		}
	}

	on_reverse_demand = {
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
		
		attacker = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_penalty_massive_effect = yes
			}
		}
		
		attacker = {
			transfer_scaled_wealth = {
				to = defender
				value = 2.0
			}
			
			prestige = -1000
		}
		
		defender = {
			participation_scaled_prestige = 2000
		}
		
		any_defender = {
			limit = { NOT = { character = defender } }
			
			hidden_tooltip = {
				participation_scaled_prestige = 2000
			}
		}
		
		# 1: Take the attacker's vassals
		attacker = {
			any_vassal = {
				limit = {
					higher_tier_than = baron
				}
				
				set_defacto_liege = defender
			}
		}
		
		# 2: Vassalize the attacker and imprison them
		ROOT = {
			e_great_wall = {
				clear_all_chinese_dynasty_flags_and_update_global_flags_effect = yes
			}
			
			e_great_wall = {
				show_scope_change = no
				unsafe_destroy_landed_title = THIS
			}
			
			set_defacto_liege = defender
			
			imprison = defender
			
			add_trait = former_eoc_trait
		}
		
		defender = {
			opinion = {
				who = attacker
				modifier = opinion_traitor
			}
		}
		
		# 3: Potential Sixteen Prefectures event
		any_attacker = {
			limit = { NOT = { character = attacker } }
			if = {
				limit = {
					independent = yes
					tier = emperor
					government = chinese_imperial_government
					
					dynasty = 1045082 # Liao Dynasty
					
					any_owned_bloodline = {
						has_bloodline_flag = liao_china
					}
				}
				
				defender = {
					character_event = { id = soh.7208 } # Sixteen Prefectures event
				}
			}
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

fang_las_rebellion = {
	name = CB_NAME_FANG_LAS_REBELLION
	war_name = WAR_NAME_FANG_LAS_REBELLION
	sprite = 9
	truce_days = 3650
	is_permanent = no # Only started from script

	allowed_to_target_tributaries = no
	can_ask_to_join_war = no
	allow_whitepeace = no # You win, or you lose
	
	attacker_can_call_allies = no # Their land would be at risk...

	can_use = {
		always = no # Should only be triggered in script
	}
	
	on_success = {
		attacker = {
			prestige = 1000
			
			vassalize_or_take_under_title = {
				title = k_liangzhe
				enemy = defender
				type = invasion
			}
			
			if = {
				limit = {
					k_liangzhe = {
						has_holder = no
					}
				}
				
				grant_title = k_liangzhe
			}
			
			else_if = {
				limit = {
					k_liangzhe = {
						holder_scope = {
							top_liege = {
								character = defender
							}
						}
					}
				}
				
				usurp_title = k_liangzhe
			}
		}
		
		defender = {
			prestige = -1000
			
			opinion = {
				who = attacker
				modifier = opinion_traitor
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				
				mandate_penalty_massive_effect = yes
			}
		}
	}
	
	on_reverse_demand = {
		defender = {
			prestige = 1000
		}
		
		attacker = {
			prestige = -1000
			
			disband_event_forces = fang_las_forces
			
			any_demesne_title = {
				usurp_title = {
					target = defender
					type = invasion
				}
			}
			
			imprison = defender
			
			reverse_opinion = {
				who = defender
				modifier = opinion_traitor
			}
		}
		
		any_attacker = {
			limit = {
				NOT = {
					character = attacker
				}
			}
			
			prestige = -1000
			
			disband_event_forces = lu_shinangs_forces
			
			imprison = defender
			
			reverse_opinion = {
				who = defender
				modifier = opinion_traitor
			}
			
			any_demesne_title = {
				usurp_title = {
					target = defender
					type = invasion
				}
			}
		}
		
		defender = {
			if = {
				limit = {
					government = chinese_imperial_government
				}
				mandate_boost_massive_effect = yes
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
