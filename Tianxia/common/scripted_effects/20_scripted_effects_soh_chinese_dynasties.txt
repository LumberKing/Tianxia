				}
			}
		}

			capital_scope = {
				county = {
					usurp_title = ROOT
				}
			}
		}
	}
	any_demesne_title = {
		limit = {
			OR = {
				tier = king
				tier = duke
			}
			controls_religion = no
		}
		unsafe_destroy_landed_title = THIS
	}
	if = {
		limit = {
			e_china = {
				holder_scope = {
					OR = {
						dynasty = ROOT
						is_close_relative = ROOT
					}
				}
			}
		}
		any_tributary = {
			decrease_tributary_tier_or_release_effect = yes
		}
	}
	else = {
		any_tributary = {
			ROOT = {
				remove_tributary = PREV
			}
		}
	}
}
