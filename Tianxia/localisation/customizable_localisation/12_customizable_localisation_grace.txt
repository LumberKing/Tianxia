defined_text = {
	name = GetAppropriateChinaName
	use_first_valid = yes
	
	text = {
		localisation_key = GETCHINATREASUREFLEET
		trigger = {
			has_landed_title = d_treasure_fleet
		}
	}
	
	text = {
		localisation_key = GETCHINATOPLIEGE
		trigger = {
			NOT = {
				has_landed_title = d_treasure_fleet
			}
			top_liege = {
				government = chinese_imperial_government
			}
		}
	}
	
	text = {
		localisation_key = GETCHINATOPLIEGESUZERAIN
		trigger = {
			NOT = {
				has_landed_title = d_treasure_fleet
			}
			top_liege = {
				NOT = {
					government = chinese_imperial_government
				}
				suzerain = {
					government = chinese_imperial_government
				}
			}
		}
	}
	
	text = {
		localisation_key = GETCHINACHINA
		trigger = {
			NOT = {
				has_landed_title = d_treasure_fleet
			}
			top_liege = {
				NOR = {
					government = chinese_imperial_government
					suzerain = {
						government = chinese_imperial_government
					}
					government = eastern_imperial_government
					government = japanese_imperial_government
					government = divine_imperial_government
				}
			}
		}
	}
	
	fallback_text = {
		localisation_key = CHINA
	}
}
